46template <unsigned ELEMENT_DIM, unsigned SPACE_DIM>
47Hdf5ToTxtConverter<ELEMENT_DIM, SPACE_DIM>::Hdf5ToTxtConverter(const FileFinder& rInputDirectory,
48 const std::string& rFileBaseName,
49 AbstractTetrahedralMesh<ELEMENT_DIM, SPACE_DIM>* pMesh)
50 : AbstractHdf5Converter<ELEMENT_DIM,SPACE_DIM>(rInputDirectory, rFileBaseName, pMesh, "txt_output", 0u)
51{
52 // Make sure that we are never trying to write from an incomplete data HDF5 file
53 assert(this->mpReader->GetNumberOfRows() == pMesh->GetNumNodes());
54
55 FileFinder output_directory(this->mRelativeSubdirectory,rInputDirectory);
56 OutputFileHandler handler(output_directory);
57
58 unsigned num_nodes = pMesh->GetNumNodes();
59 unsigned num_timesteps = this->mpReader->GetUnlimitedDimensionValues().size();
60
61 DistributedVectorFactory* p_factory = pMesh->GetDistributedVectorFactory();
62 Vec data = p_factory->CreateVec();
63 ReplicatableVector repl_data(num_nodes);
64
65 // Loop over time steps
66 for (unsigned time_step=0; time_step<num_timesteps; time_step++)
67 {
68 // Loop over variables
69 for (unsigned var_index=0; var_index<this->mNumVariables; var_index++)
70 {
71 std::string variable_name = this->mpReader->GetVariableNames()[var_index];
72
73 // Create a .txt file for this time step and this variable
74 std::stringstream file_name;
75 file_name << rFileBaseName << "_" << variable_name << "_" << time_step << ".txt";
76 out_stream p_file = handler.OpenOutputFile(file_name.str());
77
78 this->mpReader->GetVariableOverNodes(data, variable_name, time_step);
79 repl_data.ReplicatePetscVector(data);
80
81 assert(repl_data.GetSize() == num_nodes);
82
83 if (PetscTools::AmMaster())
84 {
85 for (unsigned i=0; i<num_nodes; i++)
86 {
87 *p_file << repl_data[i] << "\n";
88 }
89 }
90 p_file->close();
91 }
92 }
93
94 // Tidy up
95 PetscTools::Destroy(data);
96}

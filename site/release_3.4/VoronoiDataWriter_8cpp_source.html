50 template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
51 void VoronoiDataWriter<ELEMENT_DIM, SPACE_DIM>::Visit(MeshBasedCellPopulation<ELEMENT_DIM, SPACE_DIM>* pCellPopulation)
52 {
53  assert(SPACE_DIM==2 || SPACE_DIM==3);
54  VertexMesh<ELEMENT_DIM,SPACE_DIM>* voronoi_tesselation = pCellPopulation->GetVoronoiTessellation();
55 
56  // Loop over elements of voronoi_tesselation
57  for (typename VertexMesh<ELEMENT_DIM,SPACE_DIM>::VertexElementIterator elem_iter = voronoi_tesselation->GetElementIteratorBegin();
58  elem_iter != voronoi_tesselation->GetElementIteratorEnd();
59  ++elem_iter)
60  {
61  // Get index of this element in voronoi_tesselation
62  unsigned elem_index = elem_iter->GetIndex();
63 
64  // Get the index of the corresponding node in mrMesh
65  unsigned node_index = voronoi_tesselation->GetDelaunayNodeIndexCorrespondingToVoronoiElementIndex(elem_index);
66 
67  // Write node index and location to file
68  *this->mpOutStream << node_index << " ";
69  c_vector<double, SPACE_DIM> node_location = pCellPopulation->GetNode(node_index)->rGetLocation();
70  for (unsigned i=0; i<SPACE_DIM; i++)
71  {
72  *this->mpOutStream << node_location[i] << " ";
73  }
74 
75  double cell_volume = voronoi_tesselation->GetVolumeOfElement(elem_index);
76  double cell_surface_area = voronoi_tesselation->GetSurfaceAreaOfElement(elem_index);
77  *this->mpOutStream << cell_volume << " " << cell_surface_area << " ";
78  }
79 }

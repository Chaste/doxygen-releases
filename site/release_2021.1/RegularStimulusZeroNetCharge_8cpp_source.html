46 double RegularStimulusZeroNetCharge::GetStimulus(double time)
47 {
48  /*
49  * This covers a borderline case where start time is larger than period
50  * and period divides start time, e.g. start_time=2 period=1.
51  *
52  * fmod() will return -0 in this case and will therefore switch the stimulus on.
53  */
54  if (time < mStartTime)
55  {
56  return 0.0;
57  }
58 
59  double beatTime = fmod(time-mStartTime,mPeriod);
60 
61  if (beatTime >=0 && beatTime <= mDuration/2 && time <= mStopTime)
62  {
63  return mMagnitudeOfStimulus;
64  }
65  else if (beatTime >=0 && beatTime > mDuration/2 && beatTime <= mDuration && time <= mStopTime)
66  {
67  return -mMagnitudeOfStimulus;
68  }
69  else
70  {
71  return 0.0;
72  }
73 }

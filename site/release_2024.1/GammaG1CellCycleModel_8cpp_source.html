74void GammaG1CellCycleModel::SetG1Duration()
75{
76 if (mpCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>()
77 || mpCell->GetCellProliferativeType()->IsType<TransitCellProliferativeType>() )
78 {
79 // Generate a gamma random number with mShape and mScale
80 mG1Duration = RandomNumberGenerator::Instance()->GammaRandomDeviate(mShape, mScale);
81 }
82 else if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
83 {
84 mG1Duration = DBL_MAX;
85 }
86 else
87 {
88 NEVER_REACHED;
89 }
90}

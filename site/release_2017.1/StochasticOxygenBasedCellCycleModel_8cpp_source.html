67 void StochasticOxygenBasedCellCycleModel::GenerateStochasticG2Duration()
68 {
69  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
70 
71  double mean = AbstractPhaseBasedCellCycleModel::GetG2Duration();
72  double standard_deviation = 1.0;
73 
74  mStochasticG2Duration = p_gen->NormalRandomDeviate(mean, standard_deviation);
75 
76  // Check that the normal random deviate has not returned a small or negative G2 duration
77  if (mStochasticG2Duration < mMinimumGapDuration)
78  {
79  mStochasticG2Duration = mMinimumGapDuration;
80  }
81 }

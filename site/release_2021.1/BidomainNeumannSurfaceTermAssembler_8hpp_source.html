88 template <unsigned ELEMENT_DIM, unsigned SPACE_DIM>
89 c_vector<double, 2*ELEMENT_DIM> BidomainNeumannSurfaceTermAssembler<ELEMENT_DIM, SPACE_DIM>::ComputeVectorSurfaceTerm(
90  const BoundaryElement<ELEMENT_DIM-1,SPACE_DIM>& rSurfaceElement,
91  c_vector<double, ELEMENT_DIM>& rPhi,
92  ChastePoint<SPACE_DIM>& rX)
93 {
94  // D_times_gradu_dot_n = [D grad(u)].n, D=diffusion matrix
95  double sigma_i_times_grad_phi_i_dot_n = this->mpBoundaryConditions->GetNeumannBCValue(&rSurfaceElement, rX, 0);
96  double sigma_e_times_grad_phi_e_dot_n = this->mpBoundaryConditions->GetNeumannBCValue(&rSurfaceElement, rX, 1);
97 
98  c_vector<double, 2*ELEMENT_DIM> ret;
99  for (unsigned i=0; i<ELEMENT_DIM; i++)
100  {
101  ret(2*i) = rPhi(i)*sigma_i_times_grad_phi_i_dot_n;
102  ret(2*i+1) = rPhi(i)*(sigma_i_times_grad_phi_i_dot_n + sigma_e_times_grad_phi_e_dot_n);
103  }
104 
105  return ret;
106 }

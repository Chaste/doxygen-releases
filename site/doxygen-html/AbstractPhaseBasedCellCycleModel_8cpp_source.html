81bool AbstractPhaseBasedCellCycleModel::ReadyToDivide()
82{
83 assert(mpCell != nullptr);
84
85 if (!mReadyToDivide)
86 {
87 UpdateCellCyclePhase();
88 if ((mCurrentCellCyclePhase != G_ZERO_PHASE) &&
89 (GetAge() >= GetMDuration() + GetG1Duration() + GetSDuration() + GetG2Duration()))
90 {
91 mReadyToDivide = true;
92 }
93 }
94 return mReadyToDivide;
95}

00058 bool ChasteEllipsoid<1>::DoesContain(const ChastePoint<1>& rPointToCheck) const
00059 {
00060     if (rPointToCheck[0] < mCentre[0] - mRadii[0] - 100.0 * DBL_EPSILON ||
00061         rPointToCheck[0] > mCentre[0] + mRadii[0] + 100.0 * DBL_EPSILON )
00062     {
00063         return false;
00064     }
00065     else
00066     {
00067         return true;

00079 template <>
00080 bool ChasteEllipsoid<2>::DoesContain(const ChastePoint<2>& rPointToCheck) const
00081 {
00082     double x_distance = (rPointToCheck[0]-mCentre[0])/mRadii[0];
00083     double y_distance = (rPointToCheck[1]-mCentre[1])/mRadii[1];
00084 
00085     if ( (x_distance*x_distance + y_distance*y_distance) > (1.0 + 100.0 * DBL_EPSILON) )
00086     {
00087         return false;
00088     }
00089     else
00090     {
00091         return true;

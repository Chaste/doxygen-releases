40 void CuboidMeshConstructor<ELEMENT_DIM, SPACE_DIM>::Construct(AbstractTetrahedralMesh<ELEMENT_DIM,SPACE_DIM>& rMesh, unsigned meshRefinementNum, double meshWidth)
41 {
42  // The class has only been tested for ELEMENT_DIM == SPACE_DIM or ELEMENT_DIM == 1 && SPACE_DIM == 3
43  assert(ELEMENT_DIM == SPACE_DIM || (ELEMENT_DIM == 1 && SPACE_DIM == 3));
44 
45  mMeshWidth = meshWidth;
46  assert(meshRefinementNum < 30); // Sanity
47 
48  // Create the mesh
49  unsigned mesh_size = SmallPow(2u, meshRefinementNum+2); // number of elements in each dimension
50  double scaling = mMeshWidth/(double) mesh_size;
51  rMesh.ConstructRegularSlabMesh(scaling, mMeshWidth, mMeshWidth, mMeshWidth);
52  mNumElements = rMesh.GetNumElements();
53  mNumNodes = rMesh.GetNumNodes();
54 }

62bool LabelDependentBernoulliTrialCellCycleModel::ReadyToDivide()
63{
64 assert(mpCell != nullptr);
65
66 if (!mReadyToDivide)
67 {
68 if (GetAge() > mMinimumDivisionAge)
69 {
70 double dt = SimulationTime::Instance()->GetTimeStep();
71 if (!(mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>()))
72 {
73 RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
74 double division_probability = mDivisionProbability;
75 if (mpCell->HasCellProperty<CellLabel>())
76 {
77 division_probability = mLabelledDivisionProbability;
78 }
79 if (p_gen->ranf() < division_probability*dt)
80 {
81 mReadyToDivide = true;
82 }
83 }
84 }
85 }
86 return mReadyToDivide;
87}

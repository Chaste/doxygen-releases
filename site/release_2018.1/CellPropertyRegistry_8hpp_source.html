152 template<class SUBCLASS>
153 boost::shared_ptr<AbstractCellProperty> CellPropertyRegistry::Get()
154 {
155  boost::shared_ptr<AbstractCellProperty> p_property;
156  for (unsigned i=0; i<mCellProperties.size(); i++)
157  {
158  if (mCellProperties[i]->IsType<SUBCLASS>())
159  {
160  p_property = mCellProperties[i];
161  break;
162  }
163  }
164  if (!p_property)
165  {
166  // Create a new cell property
167  p_property.reset(new SUBCLASS);
168  mCellProperties.push_back(p_property);
169  }
170  return p_property;
171 }

48 bool AbstractSteadyStateRunner::RunToSteadyState()
49 {
50  // Get timing information from the cell stimulus (only works for regular stimuli)
51  if (!boost::dynamic_pointer_cast<RegularStimulus>(mpModel->GetStimulusFunction()))
52  {
53  EXCEPTION("Steady State approximations only work for models with RegularStimulus objects.");
54  }
55 
56  RunToSteadyStateImplementation();
57 
58  if (mNumEvaluations < mMaxNumPaces)
59  {
60  if (!mSuppressOutput)
61  {
62  std::cout << "Steady state detected after " << GetNumEvaluations() << " paces." << std::endl;
63  }
64  return true;
65  }
66  else
67  {
68  if (!mSuppressOutput)
69  {
70  std::cout << "Steady state not reached after " << GetNumEvaluations() << " paces." << std::endl;
71  }
72  WARNING("Model " << mpModel->GetSystemName() << " did not reach steady state within " << mMaxNumPaces << " paces.");
73  return false;
74  }
75 }

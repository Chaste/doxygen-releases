119template<class Archive, unsigned SPACE_DIM>
120inline void save_construct_data(
121 Archive & ar, const ChasteNodesList<SPACE_DIM> * t, const unsigned int file_version)
122{
123 const std::vector<Node<SPACE_DIM>*>& node_list = t->rGetNodesList();
124
125 // Archive the size first
126 unsigned size = t->GetSize();
127 ar & size;
128
129 // The ChastePoints have to be at unique addresses, otherwise Boost thinks that the first archive is sufficient
130 std::vector <ChastePoint<SPACE_DIM>* > point_list;
131
132 for (unsigned i = 0; i < node_list.size(); i++)
133 {
134 c_vector<double, SPACE_DIM> loc = node_list[i]->rGetLocation();
135 point_list.push_back(new ChastePoint<SPACE_DIM>(loc));
136 ar & point_list[i];
137 unsigned index = node_list[i]->GetIndex();
138 ar & index;
139 }
140
141 // Clean memory
142 for (unsigned i = 0; i < point_list.size(); i++)
143 {
144 delete point_list[i];
145 }
146}
147
152template<class Archive, unsigned SPACE_DIM>
153inline void load_construct_data(
154 Archive & ar, ChasteNodesList<SPACE_DIM> * t, const unsigned int file_version)
155{
156 // Unarchive the size first
157 unsigned size;
158 ar & size;
159
160 // Rebuild the node list based on the unarchived points and indices
161 std::vector<Node<SPACE_DIM>* > node_list;
162 for (unsigned i = 0; i < size; i++)
163 {
164 ChastePoint<SPACE_DIM>* p_point;
165 unsigned index;
166 ar & p_point;
167 ar & index;
168
169 Node<SPACE_DIM>* p_node = new Node<SPACE_DIM>( index, *(p_point));
170 node_list.push_back(p_node);
171
172 delete p_point; // not needed any longer
173 }
174
175 ::new(t)ChasteNodesList<SPACE_DIM>(node_list, true);
176}

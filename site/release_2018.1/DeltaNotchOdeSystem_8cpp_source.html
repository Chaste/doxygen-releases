69 void DeltaNotchOdeSystem::EvaluateYDerivatives(double time, const std::vector<double>& rY, std::vector<double>& rDY)
70 {
71  double notch = rY[0];
72  double delta = rY[1];
73  double mean_delta = this->mParameters[0]; // Shorthand for "this->mParameter("Mean Delta");"
74 
75  // The next two lines define the ODE system by Collier et al. (1996)
76  rDY[0] = mean_delta*mean_delta/(0.01 + mean_delta*mean_delta) - notch; // d[Notch]/dt
77  rDY[1] = 1.0/(1.0 + 100.0*notch*notch) - delta; // d[Delta]/dt
78 }

48template <class CELL, unsigned ELEMENT_DIM, unsigned SPACE_DIM = ELEMENT_DIM>
49class GeneralPlaneStimulusCellFactory : public PlaneStimulusCellFactory<CELL,ELEMENT_DIM,SPACE_DIM>

51
52public:
61 GeneralPlaneStimulusCellFactory(unsigned numEleAcross, double meshWidth, double stimulusMagnitude=-1e7, double stimulusDuration=0.5)
62 : PlaneStimulusCellFactory<CELL,ELEMENT_DIM, SPACE_DIM>(stimulusMagnitude,stimulusDuration) // These values are overridden below anyway.
63 {
64 stimulusMagnitude*=numEleAcross/(64.0);
65 // ELEMENT_DIM==1 Justification: elements go half size with each refinement
66 // ELEMENT_DIM==2 Justification: Triangles go quarter size with each refinement, but there are twice as many nodes on boundary
67 // ELEMENT_DIM==3 Hypothesis: Triangles go eighth size with each refinement, but there are four-times as many nodes on boundary
68 stimulusMagnitude*=meshWidth/(0.2);
69
70 this->mpStimulus.reset(new SimpleStimulus(stimulusMagnitude, stimulusDuration));
71 LOG(1, "Defined a GeneralPlaneStimulusCellFactory<"<<SPACE_DIM<<"> with SimpleStimulus("<<stimulusMagnitude<<","<< stimulusDuration<< ")\n");
72 }
73};

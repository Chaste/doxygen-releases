43RegularStimulus::RegularStimulus(double magnitudeOfStimulus, double duration, double period, double startTime, double stopTime)
44{
45 assert(period > 0);
46 assert(period >= duration);
47
48 mMagnitudeOfStimulus = magnitudeOfStimulus;
49 mDuration = duration;
50 mPeriod = period;
51 mStartTime = startTime;
52 mStopTime = stopTime;
53
54 //Swell duration to avoid rounding issues
55 mDuration += period*DBL_EPSILON;
56}

58double RegularStimulus::GetStimulus(double time)
59{
60 /*
61 * This covers a borderline case where start time is larger than period
62 * and period divides start time, e.g. start_time=2 period=1.
63 *
64 * fmod() will return -0 in this case and will therefore switch the stimulus on.
65 */
66 if (time < mStartTime)
67 {
68 return 0.0;
69 }
70
71 double beatTime = fmod(time-mStartTime, mPeriod);
72
73 if (beatTime >=0 && beatTime <= mDuration && time <= mStopTime)
74 {
75 return mMagnitudeOfStimulus;
76 }
77 else
78 {
79 return 0.0;
80 }
81}

65 c_matrix<double,SPACE_DIM,SPACE_DIM>& rGetModifiedConductivityTensor(unsigned elementIndex,
66 const c_matrix<double,SPACE_DIM,SPACE_DIM>& rOriginalConductivity,
67 unsigned domainIndex)
68 {
69 // Have we got space for this domain?
70 if (mCache.size() <= domainIndex)
71 {
72 // Not a pretty line! Initialises every new entry with an UNSIGNED_UNSET and a zero matrix.
73 mCache.resize(domainIndex+1, std::pair<unsigned, c_matrix<double,SPACE_DIM,SPACE_DIM> >
74 (UNSIGNED_UNSET, zero_matrix<double>(SPACE_DIM,SPACE_DIM)));
75 }
76 // Is this not the same element as last time?
77 if (mCache[domainIndex].first != elementIndex)
78 {
79 mCache[domainIndex].first = elementIndex;
80 mCache[domainIndex].second = rCalculateModifiedConductivityTensor(elementIndex, rOriginalConductivity, domainIndex);
81 }
82 // Return cached tensor
83 return mCache[domainIndex].second;
84 }

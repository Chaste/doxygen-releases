38 void RungeKutta4IvpOdeSolver::CalculateNextYValue(AbstractOdeSystem* pAbstractOdeSystem,
39  double timeStep,
40  double time,
41  std::vector<double>& rCurrentYValues,
42  std::vector<double>& rNextYValues)
43 {
44  /*
45  * Apply Runge-Kutta 4th order method for each timestep in AbstractOneStepIvpSolver.
46  * Calculates a vector containing the next Y value from the current one for each
47  * equation in the system.
48  */
49 
50  const unsigned num_equations = pAbstractOdeSystem->GetNumberOfStateVariables();
51 
52  if (num_equations != k1.size())
53  {
54  k1.resize(num_equations);
55  k2.resize(num_equations);
56  k3.resize(num_equations);
57  k4.resize(num_equations);
58  yki.resize(num_equations);
59  }
60 
61  std::vector<double>& dy = rNextYValues; // re-use memory
62 
63  pAbstractOdeSystem->EvaluateYDerivatives(time, rCurrentYValues, dy);
64  for (unsigned i=0; i<num_equations; i++)
65  {
66  k1[i] = timeStep*dy[i];
67  yki[i] = rCurrentYValues[i] + 0.5*k1[i];
68  }
69 
70  pAbstractOdeSystem->EvaluateYDerivatives(time+0.5*timeStep, yki, dy);
71  for (unsigned i=0; i<num_equations; i++)
72  {
73  k2[i] = timeStep*dy[i];
74  yki[i] = rCurrentYValues[i] + 0.5*k2[i];
75  }
76 
77  pAbstractOdeSystem->EvaluateYDerivatives(time+0.5*timeStep, yki, dy);
78  for (unsigned i=0; i<num_equations; i++)
79  {
80  k3[i] = timeStep*dy[i];
81  yki[i] = rCurrentYValues[i] + k3[i];
82  }
83 
84  pAbstractOdeSystem->EvaluateYDerivatives(time+timeStep, yki, dy);
85  for (unsigned i=0; i<num_equations; i++)
86  {
87  k4[i] = timeStep*dy[i];
88  rNextYValues[i] = rCurrentYValues[i] + (k1[i]+2*k2[i]+2*k3[i]+k4[i])/6.0;
89  }
90 }

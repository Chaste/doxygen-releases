106 double AbstractCardiacCellInterface::GetIntracellularAreaStimulus(double time)
107 {
108  double stim;
109  if (mIsUsedInTissue)
110  {
111  // Convert from uA/cm^3 to uA/cm^2 by dividing by Am
112  stim = GetIntracellularStimulus(time) / HeartConfig::Instance()->GetSurfaceAreaToVolumeRatio();
113  }
114  else
115  {
116  stim = GetIntracellularStimulus(time);
117  }
118  return stim;
119 }

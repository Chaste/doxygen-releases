40CombinedOdeSystem::CombinedOdeSystem(std::vector<AbstractOdeSystem*> odeSystems)
41 : AbstractOdeSystem(0) // will be set properly below
42{
43 mOdeSystems = odeSystems;
44 for (unsigned i=0; i<odeSystems.size(); i++)
45 {
46 mNumberOfStateVariables += odeSystems[i]->GetNumberOfStateVariables();
47 }
48 mpSystemInfo = CombinedOdeSystemInformation::Instance(odeSystems);
49 ResetToInitialConditions();
50
51 // Set up working memory
52 unsigned num_systems = odeSystems.size();
53 mWorkingStateVars.resize(num_systems);
54 mWorkingDerivs.resize(num_systems);
55 unsigned offset = 0;
56 for (unsigned i=0; i<num_systems; i++)
57 {
58 unsigned num_vars = odeSystems[i]->GetNumberOfStateVariables();
59 mWorkingStateVars[i].resize(num_vars);
60 mWorkingDerivs[i].resize(num_vars);
61 mOffsets.push_back(offset);
62 offset += num_vars;
63 }
64}

67void CombinedOdeSystem::Configure(
68 const std::map<unsigned, unsigned>& rVariableParameterMap,
69 AbstractOdeSystem* pVariableOdeSystem,
70 AbstractOdeSystem* pParameterOdeSystem)
71{
72 struct VariableParameterMap new_map;
73 new_map.theMap = rVariableParameterMap;
74 unsigned var_system_index = 0;
75 while (var_system_index < mOdeSystems.size() && mOdeSystems[var_system_index] != pVariableOdeSystem)
76 {
77 ++var_system_index;
78 }
79 new_map.pVariableOdeSystemIndex = var_system_index;
80 new_map.pParameterOdeSystem = pParameterOdeSystem;
81 mVariableParameterMaps.push_back(new_map);
82}

85void CombinedOdeSystem::EvaluateYDerivatives(
86 double time,
87 const std::vector<double>& rY,
88 std::vector<double>& rDY)
89{
90 // Copy rY to subsystems
91 for (unsigned i=0; i<mOdeSystems.size(); i++)
92 {
93 unsigned offset = mOffsets[i];
94 for (unsigned j=0; j<mOdeSystems[i]->GetNumberOfStateVariables(); j++)
95 {
96 mWorkingStateVars[i][j] = rY[offset + j];
97 }
98 }
99
100 // Set parameter values
101 for (unsigned i=0; i<mVariableParameterMaps.size(); i++)
102 {
103 std::map<unsigned, unsigned>& r_var_param_map = mVariableParameterMaps[i].theMap;
104 // Iterate through map
105 for (std::map<unsigned, unsigned>::iterator iter = r_var_param_map.begin();
106 iter != r_var_param_map.end();
107 ++iter)
108 {
109 double value = mWorkingStateVars[mVariableParameterMaps[i].pVariableOdeSystemIndex][iter->first];
110 mVariableParameterMaps[i].pParameterOdeSystem->SetParameter(iter->second, value);
111 }
112 }
113
114 // Call EvaluateYDerivatives on subsystems
115 for (unsigned i=0; i<mOdeSystems.size(); i++)
116 {
117 mOdeSystems[i]->EvaluateYDerivatives(time, mWorkingStateVars[i], mWorkingDerivs[i]);
118 }
119
120 // Copy derivatives to rDY
121 for (unsigned i=0; i<mOdeSystems.size(); i++)
122 {
123 unsigned offset = mOffsets[i];
124 for (unsigned j=0; j<mOdeSystems[i]->GetNumberOfStateVariables(); j++)
125 {
126 rDY[offset + j] = mWorkingDerivs[i][j];
127 }
128 }
129}

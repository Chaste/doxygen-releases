45 void StochasticOxygenBasedCellCycleModel::GenerateStochasticG2Duration()
46 {
47  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
48 
49  double mean = AbstractCellCycleModel::GetG2Duration();
50  double standard_deviation = 1.0;
51 
52  mStochasticG2Duration = p_gen->NormalRandomDeviate(mean, standard_deviation);
53 
54  // Check that the normal random deviate has not returned a small or negative G2 duration
55  if (mStochasticG2Duration < mMinimumGapDuration)
56  {
57  mStochasticG2Duration = mMinimumGapDuration;
58  }
59 }

84 AbstractCellCycleModel* StochasticOxygenBasedCellCycleModel::CreateCellCycleModel()
85 {
86  // Create a new cell-cycle model
87  StochasticOxygenBasedCellCycleModel* p_model = new StochasticOxygenBasedCellCycleModel();
88 
89  /*
90  * Set each member variable of the new cell-cycle model that inherits
91  * its value from the parent.
92  *
93  * Note 1: some of the new cell-cycle model's member variables (namely
94  * mBirthTime, mCurrentCellCyclePhase, mReadyToDivide, mTimeSpentInG1Phase,
95  * mCurrentHypoxicDuration, mCurrentHypoxiaOnsetTime) will already have been
96  * correctly initialized in its constructor.
97  *
98  * Note 2: one or more of the new cell-cycle model's member variables
99  * may be set/overwritten as soon as InitialiseDaughterCell() is called on
100  * the new cell-cycle model.
101  */
102  p_model->SetBirthTime(mBirthTime);
103  p_model->SetDimension(mDimension);
104  p_model->SetMinimumGapDuration(mMinimumGapDuration);
105  p_model->SetStemCellG1Duration(mStemCellG1Duration);
106  p_model->SetTransitCellG1Duration(mTransitCellG1Duration);
107  p_model->SetSDuration(mSDuration);
108  p_model->SetG2Duration(mG2Duration);
109  p_model->SetMDuration(mMDuration);
110  p_model->SetHypoxicConcentration(mHypoxicConcentration);
111  p_model->SetQuiescentConcentration(mQuiescentConcentration);
112  p_model->SetCriticalHypoxicDuration(mCriticalHypoxicDuration);
113  p_model->SetCurrentHypoxiaOnsetTime(mCurrentHypoxiaOnsetTime);
114 
115  return p_model;
116 }

65double AbstractWntOdeBasedCellCycleModel::GetWntLevel() const
66{
67 assert(mpCell != nullptr);
68 double level = 0;
69
70 switch (mDimension)
71 {
72 case 1:
73 {
74 const unsigned DIM = 1;
75 level = WntConcentration<DIM>::Instance()->GetWntLevel(mpCell);
76 break;
77 }
78 case 2:
79 {
80 const unsigned DIM = 2;
81 level = WntConcentration<DIM>::Instance()->GetWntLevel(mpCell);
82 break;
83 }
84 case 3:
85 {
86 const unsigned DIM = 3;
87 level = WntConcentration<DIM>::Instance()->GetWntLevel(mpCell);
88 break;
89 }
90 default:
91 NEVER_REACHED;
92 }
93
94 return level;
95}

97void AbstractWntOdeBasedCellCycleModel::ResetForDivision()
98{
99 AbstractOdeBasedPhaseBasedCellCycleModel::ResetForDivision();
100
101 assert(mpOdeSystem != nullptr);
102
103 // This model needs the protein concentrations and phase resetting to G0/G1.
104 // Keep the Wnt pathway in the same state but reset the cell cycle part
105 // Cell cycle is proteins 0 to 4 (first 5 ODEs)
106 std::vector<double> init_conds = mpOdeSystem->GetInitialConditions();
107 for (unsigned i=0; i<5; i++)
108 {
109 mpOdeSystem->rGetStateVariables()[i] = init_conds[i];
110 }
111}

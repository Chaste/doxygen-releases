40void FibreConverter::Convert(const FileFinder& rSourceFile, const std::string& rOutputFolderName)
41{
42 const std::string leaf_name = rSourceFile.GetLeafNameNoExtension();
43 FileFinder ortho_finder(leaf_name + ".ortho", rSourceFile.GetParent());
44 if (ortho_finder.Exists())
45 {
46 FibreReader<3> ortho_reader(ortho_finder, ORTHO);
47 std::vector< c_vector<double, 3> > fibres;
48 std::vector< c_vector<double, 3> > second;
49 std::vector< c_vector<double, 3> > third;
50 ortho_reader.GetAllOrtho(fibres, second, third);
51 //Write binary file
52 FibreWriter<3> fibre_writer(rOutputFolderName, leaf_name+"_bin", false);
53 fibre_writer.SetWriteFileAsBinary();
54 fibre_writer.WriteAllOrtho(fibres, second, third);
55 }
56
57 FileFinder axi_finder(leaf_name + ".axi", rSourceFile.GetParent());
58 if (axi_finder.Exists())
59 {
60 FibreReader<3> axi_reader(axi_finder, AXISYM);
61 std::vector< c_vector<double, 3> > fibres;
62 axi_reader.GetAllAxi(fibres);
63 //Write binary file
64 FibreWriter<3> fibre_writer(rOutputFolderName, leaf_name + "_bin", false);
65 fibre_writer.SetWriteFileAsBinary();
66 fibre_writer.WriteAllAxi(fibres);
67 }
68}

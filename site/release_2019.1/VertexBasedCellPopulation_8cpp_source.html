51 template<unsigned DIM>
52 VertexBasedCellPopulation<DIM>::VertexBasedCellPopulation(MutableVertexMesh<DIM, DIM>& rMesh,
53  std::vector<CellPtr>& rCells,
54  bool deleteMesh,
55  bool validate,
56  const std::vector<unsigned> locationIndices)
57  : AbstractOffLatticeCellPopulation<DIM>(rMesh, rCells, locationIndices),
58  mDeleteMesh(deleteMesh),
59  mOutputCellRearrangementLocations(true),
60  mRestrictVertexMovement(true)
61 {
62  mpMutableVertexMesh = static_cast<MutableVertexMesh<DIM, DIM>* >(&(this->mrMesh));
63  mpVertexBasedDivisionRule.reset(new ShortAxisVertexBasedDivisionRule<DIM>());
64 
65  // If no location indices are specified, associate with elements from the mesh (assumed to be sequentially ordered).
66  std::list<CellPtr>::iterator it = this->mCells.begin();
67  for (unsigned i=0; it != this->mCells.end(); ++it, ++i)
68  {
69  unsigned index = locationIndices.empty() ? i : locationIndices[i]; // assume that the ordering matches
70  AbstractCellPopulation<DIM, DIM>::AddCellUsingLocationIndex(index,*it);
71  }
72 
73  // Check each element has only one cell attached
74  if (validate)
75  {
76  Validate();
77  }
78 }

99 template<unsigned DIM>
100 double VertexBasedCellPopulation<DIM>::GetDampingConstant(unsigned nodeIndex)
101 {
102  // Take the average of the cells containing this vertex
103  double average_damping_constant = 0.0;
104 
105  std::set<unsigned> containing_elements = GetNode(nodeIndex)->rGetContainingElementIndices();
106 
107  unsigned num_containing_elements = containing_elements.size();
108  if (num_containing_elements == 0)
109  {
110  EXCEPTION("At time " << SimulationTime::Instance()->GetTime() << ", Node " << nodeIndex << " is not contained in any elements, so GetDampingConstant() returns zero");
111  }
112 
113  double temp = 1.0/((double) num_containing_elements);
114  for (std::set<unsigned>::iterator iter = containing_elements.begin();
115  iter != containing_elements.end();
116  ++iter)
117  {
118  CellPtr p_cell = this->GetCellUsingLocationIndex(*iter);
119  bool cell_is_wild_type = p_cell->GetMutationState()->IsType<WildTypeCellMutationState>();
120 
121  if (cell_is_wild_type)
122  {
123  average_damping_constant += this->GetDampingConstantNormal()*temp;
124  }
125  else
126  {
127  average_damping_constant += this->GetDampingConstantMutant()*temp;
128  }
129  }
130 
131  return average_damping_constant;
132 }

201 template<unsigned DIM>
202 CellPtr VertexBasedCellPopulation<DIM>::AddCell(CellPtr pNewCell, CellPtr pParentCell)
203 {
204  // Get the element associated with this cell
205  VertexElement<DIM, DIM>* p_element = GetElementCorrespondingToCell(pParentCell);
206 
207  // Get the orientation of division
208  c_vector<double, DIM> division_vector = mpVertexBasedDivisionRule->CalculateCellDivisionVector(pParentCell, *this);
209 
210  // Divide the element
211  unsigned new_element_index = mpMutableVertexMesh->DivideElementAlongGivenAxis(p_element, division_vector, true);
212 
213  // Associate the new cell with the element
214  this->mCells.push_back(pNewCell);
215 
216  // Update location cell map
217  CellPtr p_created_cell = this->mCells.back();
218  this->SetCellUsingLocationIndex(new_element_index,p_created_cell);
219  this->mCellLocationMap[p_created_cell.get()] = new_element_index;
220 
221  return p_created_cell;
222 }

224 template<unsigned DIM>
225 unsigned VertexBasedCellPopulation<DIM>::RemoveDeadCells()
226 {
227  unsigned num_removed = 0;
228 
229  for (std::list<CellPtr>::iterator it = this->mCells.begin();
230  it != this->mCells.end();
231  )
232  {
233  if ((*it)->IsDead())
234  {
235  // Count the cell as dead
236  num_removed++;
237 
238  // Remove the element from the mesh if it is not deleted yet
240  if (!(this->GetElement(this->GetLocationIndexUsingCell((*it)))->IsDeleted()))
241  {
242  // This warning relies on the fact that there is only one other possibility for
243  // vertex elements to be marked as deleted: a T2 swap
244  WARN_ONCE_ONLY("A Cell is removed without performing a T2 swap. This could leave a void in the mesh.");
245  mpMutableVertexMesh->DeleteElementPriorToReMesh(this->GetLocationIndexUsingCell((*it)));
246  }
247 
248  // Delete the cell
249  it = this->mCells.erase(it);
250  }
251  else
252  {
253  ++it;
254  }
255  }
256  return num_removed;
257 }

259 template<unsigned DIM>
260 void VertexBasedCellPopulation<DIM>::CheckForStepSizeException(unsigned nodeIndex, c_vector<double,DIM>& rDisplacement, double dt)
261 {
262  double length = norm_2(rDisplacement);
263 
264  if(mRestrictVertexMovement)
265  {
266  if (length > 0.5*mpMutableVertexMesh->GetCellRearrangementThreshold())
267  {
268  rDisplacement *= 0.5*mpMutableVertexMesh->GetCellRearrangementThreshold()/length;
269 
270  std::ostringstream message;
271  message << "Vertices are moving more than half the CellRearrangementThreshold. This could cause elements to become inverted ";
272  message << "so the motion has been restricted. Use a smaller timestep to avoid these warnings.";
273 
274  double suggested_step = 0.95*dt*((0.5*mpMutableVertexMesh->GetCellRearrangementThreshold())/length);
275 
276  // The first time we see this behaviour, throw a StepSizeException, but not more than once
277  if(mThrowStepSizeException)
278  {
279  mThrowStepSizeException = false;
280  throw StepSizeException(suggested_step, message.str(), false);
281  }
282  }
283  }

292 template<unsigned DIM>
293 void VertexBasedCellPopulation<DIM>::Update(bool hasHadBirthsOrDeaths)
294 {
295  VertexElementMap element_map(mpMutableVertexMesh->GetNumAllElements());
296  mpMutableVertexMesh->ReMesh(element_map);
297 
298  if (!element_map.IsIdentityMap())
299  {
300  // Fix up the mappings between CellPtrs and VertexElements
302  std::map<Cell*, unsigned> old_map = this->mCellLocationMap;
303 
304  this->mCellLocationMap.clear();
305  this->mLocationCellMap.clear();
306 
307  for (std::list<CellPtr>::iterator cell_iter = this->mCells.begin();
308  cell_iter != this->mCells.end();
309  ++cell_iter)
310  {
311  // The cell vector should only ever contain living cells
312  unsigned old_elem_index = old_map[(*cell_iter).get()];
313  assert(!element_map.IsDeleted(old_elem_index));
314 
315  unsigned new_elem_index = element_map.GetNewIndex(old_elem_index);
316  this->SetCellUsingLocationIndex(new_elem_index, *cell_iter);
317  }
318 
319  // Check that each VertexElement has only one CellPtr associated with it in the updated cell population
320  Validate();
321  }
322 
323  element_map.ResetToIdentity();
324 }

326 template<unsigned DIM>
327 void VertexBasedCellPopulation<DIM>::Validate()
328 {
329  // Check each element has only one cell attached
330  std::vector<unsigned> validated_element = std::vector<unsigned>(this->GetNumElements(), 0);
331  for (typename AbstractCellPopulation<DIM>::Iterator cell_iter = this->Begin();
332  cell_iter != this->End();
333  ++cell_iter)
334  {
335  unsigned elem_index = this->GetLocationIndexUsingCell(*cell_iter);
336  validated_element[elem_index]++;
337  }
338 
339  for (unsigned i=0; i<validated_element.size(); i++)
340  {
341  if (validated_element[i] == 0)
342  {
343  EXCEPTION("At time " << SimulationTime::Instance()->GetTime() <<", Element " << i << " does not appear to have a cell associated with it");
344  }
345 
346  if (validated_element[i] > 1)
347  {
348  // This should never be reached as you can only set one cell per element index
349  EXCEPTION("At time " << SimulationTime::Instance()->GetTime() <<", Element " << i << " appears to have " << validated_element[i] << " cells associated with it");
350  }
351  }
352 }

396 template<unsigned DIM>
397 void VertexBasedCellPopulation<DIM>::WriteVtkResultsToFile(const std::string& rDirectory)
398 {
399 #ifdef CHASTE_VTK
400 
401  // Create mesh writer for VTK output
402  VertexMeshWriter<DIM, DIM> mesh_writer(rDirectory, "results", false);
403 
404  // Iterate over any cell writers that are present
405  unsigned num_cells = this->GetNumAllCells();
406  for (typename std::vector<boost::shared_ptr<AbstractCellWriter<DIM, DIM> > >::iterator cell_writer_iter = this->mCellWriters.begin();
407  cell_writer_iter != this->mCellWriters.end();
408  ++cell_writer_iter)
409  {
410  // Create vector to store VTK cell data
411  std::vector<double> vtk_cell_data(num_cells);
412 
413  // Iterate over vertex elements ///\todo #2512 - replace with loop over cells
414  for (typename VertexMesh<DIM,DIM>::VertexElementIterator elem_iter = mpMutableVertexMesh->GetElementIteratorBegin();
415  elem_iter != mpMutableVertexMesh->GetElementIteratorEnd();
416  ++elem_iter)
417  {
418  // Get index of this element in the vertex mesh
419  unsigned elem_index = elem_iter->GetIndex();
420 
421  // Get the cell corresponding to this element
422  CellPtr p_cell = this->GetCellUsingLocationIndex(elem_index);
423  assert(p_cell);
424 
425  // Populate the vector of VTK cell data
426  vtk_cell_data[elem_index] = (*cell_writer_iter)->GetCellDataForVtkOutput(p_cell, this);
427  }
428 
429  mesh_writer.AddCellData((*cell_writer_iter)->GetVtkCellDataName(), vtk_cell_data);
430  }
431 
432  // When outputting any CellData, we assume that the first cell is representative of all cells
433  unsigned num_cell_data_items = this->Begin()->GetCellData()->GetNumItems();
434  std::vector<std::string> cell_data_names = this->Begin()->GetCellData()->GetKeys();
435 
436  std::vector<std::vector<double> > cell_data;
437  for (unsigned var=0; var<num_cell_data_items; var++)
438  {
439  std::vector<double> cell_data_var(num_cells);
440  cell_data.push_back(cell_data_var);
441  }
442 
443  // Loop over vertex elements ///\todo #2512 - replace with loop over cells
444  for (typename VertexMesh<DIM,DIM>::VertexElementIterator elem_iter = mpMutableVertexMesh->GetElementIteratorBegin();
445  elem_iter != mpMutableVertexMesh->GetElementIteratorEnd();
446  ++elem_iter)
447  {
448  // Get index of this element in the vertex mesh
449  unsigned elem_index = elem_iter->GetIndex();
450 
451  // Get the cell corresponding to this element
452  CellPtr p_cell = this->GetCellUsingLocationIndex(elem_index);
453  assert(p_cell);
454 
455  for (unsigned var=0; var<num_cell_data_items; var++)
456  {
457  cell_data[var][elem_index] = p_cell->GetCellData()->GetItem(cell_data_names[var]);
458  }
459  }
460  for (unsigned var=0; var<num_cell_data_items; var++)
461  {
462  mesh_writer.AddCellData(cell_data_names[var], cell_data[var]);
463  }
464 
465  unsigned num_timesteps = SimulationTime::Instance()->GetTimeStepsElapsed();
466  std::stringstream time;
467  time << num_timesteps;
468 
469  mesh_writer.WriteVtkUsingMesh(*mpMutableVertexMesh, time.str());
470 
471  *(this->mpVtkMetaFile) << " <DataSet timestep=\"";
472  *(this->mpVtkMetaFile) << num_timesteps;
473  *(this->mpVtkMetaFile) << "\" group=\"\" part=\"0\" file=\"results_";
474  *(this->mpVtkMetaFile) << num_timesteps;
475  *(this->mpVtkMetaFile) << ".vtu\"/>\n";
476 #endif //CHASTE_VTK
477 }

596 template<unsigned DIM>
597 TetrahedralMesh<DIM, DIM>* VertexBasedCellPopulation<DIM>::GetTetrahedralMeshForPdeModifier()
598 {
599  // This method only works in 2D sequential
600  assert(DIM == 2); // LCOV_EXCL_LINE - disappears at compile time.
601  assert(PetscTools::IsSequential());
602 
603  unsigned num_vertex_nodes = mpMutableVertexMesh->GetNumNodes();
604  unsigned num_vertex_elements = mpMutableVertexMesh->GetNumElements();
605 
606  std::string mesh_file_name = "mesh";
607 
608  // Get a unique temporary foldername
609  std::stringstream pid;
610  pid << getpid();
611  OutputFileHandler output_file_handler("2D_temporary_tetrahedral_mesh_" + pid.str());
612  std::string output_dir = output_file_handler.GetOutputDirectoryFullPath();
613 
614  // Compute the number of nodes in the TetrahedralMesh
615  unsigned num_tetrahedral_nodes = num_vertex_nodes + num_vertex_elements;
616 
617  // Write node file
618  out_stream p_node_file = output_file_handler.OpenOutputFile(mesh_file_name+".node");
619  (*p_node_file) << std::scientific;
620  (*p_node_file) << std::setprecision(20);
621  (*p_node_file) << num_tetrahedral_nodes << "\t2\t0\t1" << std::endl;
622 
623  // Begin by writing each node in the VertexMesh
624  for (unsigned node_index=0; node_index<num_vertex_nodes; node_index++)
625  {
626  Node<DIM>* p_node = mpMutableVertexMesh->GetNode(node_index);
627 
629  unsigned index = p_node->GetIndex();
630  const c_vector<double, DIM>& r_location = p_node->rGetLocation();
631  unsigned is_boundary_node = p_node->IsBoundaryNode() ? 1 : 0;
632 
633  (*p_node_file) << index << "\t" << r_location[0] << "\t" << r_location[1] << "\t" << is_boundary_node << std::endl;
634  }
635 
636  // Now write an additional node at each VertexElement's centroid
637  unsigned num_tetrahedral_elements = 0;
638  for (unsigned vertex_elem_index=0; vertex_elem_index<num_vertex_elements; vertex_elem_index++)
639  {
640  unsigned index = num_vertex_nodes + vertex_elem_index;
641 
642  c_vector<double, DIM> location = mpMutableVertexMesh->GetCentroidOfElement(vertex_elem_index);
643 
644  // Any node located at a VertexElement's centroid will not be a boundary node
645  unsigned is_boundary_node = 0;
646  (*p_node_file) << index << "\t" << location[0] << "\t" << location[1] << "\t" << is_boundary_node << std::endl;
647 
648  // Also keep track of how many tetrahedral elements there will be
649  num_tetrahedral_elements += mpMutableVertexMesh->GetElement(vertex_elem_index)->GetNumNodes();
650  }
651  p_node_file->close();
652 
653  // Write element file
654  out_stream p_elem_file = output_file_handler.OpenOutputFile(mesh_file_name+".ele");
655  (*p_elem_file) << std::scientific;
656  (*p_elem_file) << num_tetrahedral_elements << "\t3\t0" << std::endl;
657 
658  std::set<std::pair<unsigned, unsigned> > tetrahedral_edges;
659 
660  unsigned tetrahedral_elem_index = 0;
661  for (unsigned vertex_elem_index=0; vertex_elem_index<num_vertex_elements; vertex_elem_index++)
662  {
663  VertexElement<DIM, DIM>* p_vertex_element = mpMutableVertexMesh->GetElement(vertex_elem_index);
664 
665  // Iterate over nodes owned by this VertexElement
666  unsigned num_nodes_in_vertex_element = p_vertex_element->GetNumNodes();
667  for (unsigned local_index=0; local_index<num_nodes_in_vertex_element; local_index++)
668  {
669  unsigned node_0_index = p_vertex_element->GetNodeGlobalIndex(local_index);
670  unsigned node_1_index = p_vertex_element->GetNodeGlobalIndex((local_index+1)%num_nodes_in_vertex_element);
671  unsigned node_2_index = num_vertex_nodes + vertex_elem_index;
672 
673  (*p_elem_file) << tetrahedral_elem_index++ << "\t" << node_0_index << "\t" << node_1_index << "\t" << node_2_index << std::endl;
674 
675  // Add edges to the set if they are not already present
676  std::pair<unsigned, unsigned> edge_0 = this->CreateOrderedPair(node_0_index, node_1_index);
677  std::pair<unsigned, unsigned> edge_1 = this->CreateOrderedPair(node_1_index, node_2_index);
678  std::pair<unsigned, unsigned> edge_2 = this->CreateOrderedPair(node_2_index, node_0_index);
679 
680  tetrahedral_edges.insert(edge_0);
681  tetrahedral_edges.insert(edge_1);
682  tetrahedral_edges.insert(edge_2);
683  }
684  }
685  p_elem_file->close();
686 
687  // Write edge file
688  out_stream p_edge_file = output_file_handler.OpenOutputFile(mesh_file_name+".edge");
689  (*p_edge_file) << std::scientific;
690  (*p_edge_file) << tetrahedral_edges.size() << "\t1" << std::endl;
691 
692  unsigned edge_index = 0;
693  for (std::set<std::pair<unsigned, unsigned> >::iterator edge_iter = tetrahedral_edges.begin();
694  edge_iter != tetrahedral_edges.end();
695  ++edge_iter)
696  {
697  std::pair<unsigned, unsigned> this_edge = *edge_iter;
698 
699  // To be a boundary edge both nodes need to be boundary nodes.
700  bool is_boundary_edge = false;
701  if (this_edge.first < mpMutableVertexMesh->GetNumNodes() &&
702  this_edge.second < mpMutableVertexMesh->GetNumNodes())
703  {
704  is_boundary_edge = (mpMutableVertexMesh->GetNode(this_edge.first)->IsBoundaryNode() &&
705  mpMutableVertexMesh->GetNode(this_edge.second)->IsBoundaryNode() );
706  }
707  unsigned is_boundary_edge_unsigned = is_boundary_edge ? 1 : 0;
708 
709  (*p_edge_file) << edge_index++ << "\t" << this_edge.first << "\t" << this_edge.second << "\t" << is_boundary_edge_unsigned << std::endl;
710  }
711  p_edge_file->close();
712 
713  // Having written the mesh to file, now construct it using TrianglesMeshReader
714  TetrahedralMesh<DIM, DIM>* p_mesh = new TetrahedralMesh<DIM, DIM>;
715 
716  // Nested scope so reader is destroyed before we remove the temporary files
717  {
718  TrianglesMeshReader<DIM, DIM> mesh_reader(output_dir + mesh_file_name);
719  p_mesh->ConstructFromMeshReader(mesh_reader);
720  }
721 
722  // Delete the temporary files
723  output_file_handler.FindFile("").Remove();
724 
725  // The original files have been deleted, it is better if the mesh object forgets about them
726  p_mesh->SetMeshHasChangedSinceLoading();
727 
728  return p_mesh;
729 }

731 template<unsigned DIM>
732 bool VertexBasedCellPopulation<DIM>::IsPdeNodeAssociatedWithNonApoptoticCell(unsigned pdeNodeIndex)
733 {
734  bool non_apoptotic_cell_present = true;
735 
736  if (pdeNodeIndex < this->GetNumNodes())
737  {
738  std::set<unsigned> containing_element_indices = this->GetNode(pdeNodeIndex)->rGetContainingElementIndices();
739 
740  for (std::set<unsigned>::iterator iter = containing_element_indices.begin();
741  iter != containing_element_indices.end();
742  iter++)
743  {
744  if (this->GetCellUsingLocationIndex(*iter)->template HasCellProperty<ApoptoticCellProperty>() )
745  {
746  non_apoptotic_cell_present = false;
747  break;
748  }
749  }
750  }
751  else
752  {
753  /*
754  * This node of the tetrahedral finite element mesh is in the centre of the element of the
755  * vertex-based cell population, so we can use an offset to compute which cell to interrogate.
756  */
757  non_apoptotic_cell_present = !(this->GetCellUsingLocationIndex(pdeNodeIndex - this->GetNumNodes())->template HasCellProperty<ApoptoticCellProperty>());
758  }
759 
760  return non_apoptotic_cell_present;

763 template<unsigned DIM>
764 double VertexBasedCellPopulation<DIM>::GetCellDataItemAtPdeNode(
765  unsigned pdeNodeIndex,
766  std::string& rVariableName,
767  bool dirichletBoundaryConditionApplies,
768  double dirichletBoundaryValue)
769 {
770  unsigned num_nodes = this->GetNumNodes();
771  double value = 0.0;
772 
773  // Cells correspond to nodes in the centre of the vertex element; nodes on vertices have averaged values from containing cells
774 
775  if (pdeNodeIndex >= num_nodes)
776  {
777  // Offset to relate elements in vertex mesh to nodes in tetrahedral mesh
778  assert(pdeNodeIndex-num_nodes < num_nodes);
779 
780  CellPtr p_cell = this->GetCellUsingLocationIndex(pdeNodeIndex - num_nodes);
781  value = p_cell->GetCellData()->GetItem(rVariableName);
782  }
783  else
784  {
786  if (dirichletBoundaryConditionApplies)
787  {
788  // We need to impose the Dirichlet boundaries again here as not represented in cell data
789  value = dirichletBoundaryValue;
790  }
791  else
792  {
793  assert(pdeNodeIndex < num_nodes);
794  Node<DIM>* p_node = this->GetNode(pdeNodeIndex);
795 
796  // Average over data from containing elements (cells)
797  std::set<unsigned> containing_elements = p_node->rGetContainingElementIndices();
798  for (std::set<unsigned>::iterator index_iter = containing_elements.begin();
799  index_iter != containing_elements.end();
800  ++index_iter)
801  {
802  assert(*index_iter < num_nodes);
803  CellPtr p_cell = this->GetCellUsingLocationIndex(*index_iter);
804  value += p_cell->GetCellData()->GetItem(rVariableName);
805  }
806  value /= containing_elements.size();
807  }
808  }
809 
810  return value;
811 }

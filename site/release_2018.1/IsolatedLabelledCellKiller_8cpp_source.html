50 template<unsigned DIM>
51 void IsolatedLabelledCellKiller<DIM>::CheckAndLabelCellsForApoptosisOrDeath()
52 {
53  MutableVertexMesh<DIM, DIM>& vertex_mesh = static_cast<VertexBasedCellPopulation<DIM>*>(this->mpCellPopulation)->rGetMesh();
54 
55  unsigned num_labelled_cells = this->mpCellPopulation->GetCellPropertyRegistry()->template Get<CellLabel>()->GetCellCount();
56 
57  // If there is more than one labelled cell...
58  if (num_labelled_cells > 1)
59  {
60  // Iterate over cell population
61  for (typename AbstractCellPopulation<DIM>::Iterator cell_iter = this->mpCellPopulation->Begin();
62  cell_iter != this->mpCellPopulation->End();
63  ++cell_iter)
64  {
65  // Only consider cells with the CellLabel property
66  if (cell_iter->template HasCellProperty<CellLabel>())
67  {
68  // Get the element index corresponding to this cell
69  unsigned elem_index = this->mpCellPopulation->GetLocationIndexUsingCell(*cell_iter);
70 
71  // Get the set of neighbouring element indices
72  std::set<unsigned> neighbouring_elem_indices = vertex_mesh.GetNeighbouringElementIndices(elem_index);
73 
74  // Check if any of the corresponding cells have the CellLabel property...
75  unsigned num_labelled_neighbours = 0;
76  for (std::set<unsigned>::iterator elem_iter = neighbouring_elem_indices.begin();
77  elem_iter != neighbouring_elem_indices.end();
78  ++elem_iter)
79  {
80  if (this->mpCellPopulation->GetCellUsingLocationIndex(*elem_iter)->template HasCellProperty<CellLabel>())
81  {
82  num_labelled_neighbours++;
83  }
84  }
85 
86  // ...and if none do, then kill this cell
87  if (num_labelled_neighbours == 0)
88  {
89  cell_iter->Kill();
90  }
91  }
92  }
93  }
94 }

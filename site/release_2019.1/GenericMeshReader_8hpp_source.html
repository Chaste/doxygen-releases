67 template <unsigned ELEMENT_DIM, unsigned SPACE_DIM>
68 std::shared_ptr<AbstractMeshReader<ELEMENT_DIM, SPACE_DIM> > GenericMeshReader(const std::string& rPathBaseName,
69  unsigned orderOfElements=1,
70  unsigned orderOfBoundaryElements=1,
71  bool readContainingElementsForBoundaryElements=false)
72 {
73  std::shared_ptr<AbstractMeshReader<ELEMENT_DIM, SPACE_DIM> > p_reader;
74  try
75  {
76  p_reader.reset(new TrianglesMeshReader<ELEMENT_DIM, SPACE_DIM>(rPathBaseName,
77  orderOfElements,
78  orderOfBoundaryElements,
79  readContainingElementsForBoundaryElements));
80  }
81  catch (const Exception& r_triangles_exception)
82  {
83  if (orderOfElements!=1 || orderOfBoundaryElements!=1 || readContainingElementsForBoundaryElements)
84  {
85  EXCEPTION("Quadratic meshes are only supported in Triangles format.");
86  }
87 
88  try
89  {
90  p_reader.reset(new MemfemMeshReader<ELEMENT_DIM, SPACE_DIM>(rPathBaseName));
91  }
92  catch (const Exception& r_memfem_exception)
93  {
94 #ifdef CHASTE_VTK
95  try
96  {
97  p_reader.reset(new VtkMeshReader<ELEMENT_DIM, SPACE_DIM>(rPathBaseName));
98  }
99  catch (const Exception& r_vtk_exception)
100  {
101 #endif // CHASTE_VTK
102  std::string eol("\n");
103  std::string combined_message = "Could not open appropriate mesh files for " + rPathBaseName + eol;
104  combined_message += "Triangle format: " + r_triangles_exception.GetShortMessage() + eol;
105  combined_message += "Memfem format: " + r_memfem_exception.GetShortMessage() + eol;
106 #ifdef CHASTE_VTK
107  combined_message += "Vtk format: " + r_vtk_exception.GetShortMessage() + eol;
108 #endif // CHASTE_VTK
109  EXCEPTION(combined_message);
110 #ifdef CHASTE_VTK
111  }
112 #endif // CHASTE_VTK
113  }
114  }
115  return p_reader;
116 }

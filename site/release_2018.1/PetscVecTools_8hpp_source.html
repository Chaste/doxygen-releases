150  template<size_t VECTOR_SIZE>
151  static void AddMultipleValues(Vec vector, unsigned* vectorIndices, c_vector<double, VECTOR_SIZE>& smallVector)
152  {
153  PetscInt indices_owned[VECTOR_SIZE];
154  PetscInt num_indices_owned = 0;
155  PetscInt global_row;
156  PetscInt lo, hi;
157  GetOwnershipRange(vector, lo, hi);
158 
159  for (unsigned row = 0; row<VECTOR_SIZE; row++)
160  {
161  global_row = vectorIndices[row];
162  if (global_row >=lo && global_row <hi)
163  {
164  indices_owned[num_indices_owned++] = global_row;
165  }
166  }
167 
168  if (num_indices_owned == VECTOR_SIZE)
169  {
170  VecSetValues(vector,
171  num_indices_owned,
172  indices_owned,
173  smallVector.data(),
174  ADD_VALUES);
175  }
176  else
177  {
178  /*
179  * We need continuous data, if some of the rows do not belong
180  * to the processor their values are not passed to VecSetValues.
181  */
182  double values[VECTOR_SIZE];
183  unsigned num_values_owned = 0;
184 
185  for (unsigned row = 0; row<VECTOR_SIZE; row++)
186  {
187  global_row = vectorIndices[row];
188  if (global_row >= lo && global_row < hi)
189  {
190  values[num_values_owned++] = smallVector(row);
191  }
192  }
193 
194  VecSetValues(vector,
195  num_indices_owned,
196  indices_owned,
197  values,
198  ADD_VALUES);
199  }
200  }

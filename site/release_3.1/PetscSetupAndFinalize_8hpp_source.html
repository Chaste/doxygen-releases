00064 void FpeSignalToAbort(int sig_num, siginfo_t* info, void* context )
00065 {
00066        if ( info->si_code == FPE_FLTDIV)
00067        {
00068            std::cerr<<"SIGFPE: floating point exception was divide by zero.\n";
00069        }
00070        else if ( info->si_code == FPE_FLTINV)
00071        {
00072            std::cerr<<"SIGFPE: floating point exception was an invalid operation (like 0.0/0.0).\n";
00073        }
00074        else
00075        {
00076            std::cerr<<"SIGFPE: unexpected error code.\n";
00077        }

00086     bool setUpWorld()
00087     {
00088         // The CommandLineArguments instance is filled in by the cxxtest test suite runner.
00089         CommandLineArguments* p_args = CommandLineArguments::Instance();
00090         PETSCEXCEPT(PetscInitialize(p_args->p_argc, p_args->p_argv,
00091                                     PETSC_NULL, PETSC_NULL) );
00092 
00093         // Check that the working directory is correct, or many tests will fail
00094         std::string cwd = GetCurrentWorkingDirectory() + "/";
00095         if (strcmp(cwd.c_str(), ChasteBuildRootDir()) != 0)
00096         {
00097 #define COVERAGE_IGNORE
00098             // Change directory
00099             std::cout << std::endl << "Changing directory from '" << cwd << "' to '" << ChasteBuildRootDir() << "'." << std::endl;
00100             EXPECT0(chdir, ChasteBuildRootDir());
00101             std::cout << "CWD now: " << GetCurrentWorkingDirectory() << std::endl;
00102 #undef COVERAGE_IGNORE
00103         }
00104 
00105 #ifdef TEST_FOR_FPE
00106         // Give all PETSc enabled tests the ability to trap for divide-by-zero
00107         feenableexcept(FE_DIVBYZERO | FE_INVALID );
00108         // Catch all SIGFPE signals and convert them to exceptions (before PETSc gets to them)
00109         struct sigaction sa;
00110         sa.sa_sigaction = FpeSignalToAbort;
00111         sa.sa_flags = SA_RESETHAND|SA_SIGINFO;
00112         sa.sa_restorer = 0;
00113         sigaction(SIGFPE, &sa, NULL);
00114 #endif
00115         return true;
00116     }
00118     bool tearDownWorld()
00119     {
00121         PETSCEXCEPT(PetscFinalize());
00122         return true;
00123     }
00124 };

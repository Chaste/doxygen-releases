46 AbstractCellCycleModel* StochasticDurationGenerationBasedCellCycleModel::CreateCellCycleModel()
47 {
48  // Create a new cell-cycle model
49  StochasticDurationGenerationBasedCellCycleModel* p_model = new StochasticDurationGenerationBasedCellCycleModel();
50 
51  /*
52  * Set each member variable of the new cell-cycle model that inherits
53  * its value from the parent.
54  *
55  * Note 1: some of the new cell-cycle model's member variables (namely
56  * mBirthTime, mCurrentCellCyclePhase, mReadyToDivide) will already have been
57  * correctly initialized in its constructor.
58  *
59  * Note 2: one or more of the new cell-cycle model's member variables
60  * may be set/overwritten as soon as InitialiseDaughterCell() is called on
61  * the new cell-cycle model.
62  *
63  * Note 3: the member variable mDimension remains unset, since this cell-cycle
64  * model does not need to know the spatial dimension, so if we were to call
65  * SetDimension() on the new cell-cycle model an exception would be triggered;
66  * hence we do not set this member variable.
67  */
68  p_model->SetBirthTime(mBirthTime);
69  p_model->SetMinimumGapDuration(mMinimumGapDuration);
70  p_model->SetStemCellG1Duration(mStemCellG1Duration);
71  p_model->SetTransitCellG1Duration(mTransitCellG1Duration);
72  p_model->SetSDuration(mSDuration);
73  p_model->SetG2Duration(mG2Duration);
74  p_model->SetMDuration(mMDuration);
75  p_model->SetGeneration(mGeneration);
76  p_model->SetMaxTransitGenerations(mMaxTransitGenerations);
77 
78  return p_model;
79 }

81 void StochasticDurationGenerationBasedCellCycleModel::SetG1Duration()
82 {
83  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
84 
85  assert(mpCell != NULL);
86 
87  if (mpCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>())
88  {
89  mG1Duration = GetStemCellG1Duration() + 4*p_gen->ranf(); // U[14,18] for default parameters (mStemCellG1Duration) according to Meineke
90  }
91  else if (mpCell->GetCellProliferativeType()->IsType<TransitCellProliferativeType>())
92  {
93  mG1Duration = GetTransitCellG1Duration() + 2*p_gen->ranf(); // U[4,6] for default parameters (mTransitG1CellDuration) according to Meineke
94  }
95  else if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
96  {
97  mG1Duration = DBL_MAX;
98  }
99  else
100  {
101  NEVER_REACHED;
102  }
103 }

73 void ExponentialG1GenerationalCellCycleModel::SetG1Duration()
74 {
75  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
76 
77  if (mpCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>()
78  || mpCell->GetCellProliferativeType()->IsType<TransitCellProliferativeType>() )
79  {
80  // Generate an exponential random number with mScale
81  mG1Duration = p_gen->ExponentialRandomDeviate(mRate);
82  }
83  else if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
84  {
85  mG1Duration = DBL_MAX;
86  }
87  else
88  {
89  NEVER_REACHED;
90  }
91 }

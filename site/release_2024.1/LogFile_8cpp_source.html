116void LogFile::WriteElapsedTime(std::string pre)
117{
118 double fsecs = difftime(time(nullptr),mInitTime);
119 long total_secs = static_cast<long>(floor(fsecs+0.5));
120 int total_mins = total_secs/60;
121
122 int secs = total_secs%60;
123 int mins = total_mins%60;
124 int hrs = total_mins/60;
125
126 *this << pre << "Elapsed time is: " << hrs << "h " << mins << "m " << secs << "s\n";
127}

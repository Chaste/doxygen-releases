194template<class Archive, unsigned ELEMENT_DIM, unsigned SPACE_DIM>
195inline void load_construct_data(
196 Archive & ar, MixedDimensionMesh<ELEMENT_DIM, SPACE_DIM> * t, const unsigned int file_version)
197{
198 unsigned num_procs;
199 DistributedTetrahedralMeshPartitionType::type partition_type;
200
201 ar >> num_procs;
202 ar >> partition_type;
203
204 // Invoke inplace constructor to initialise instance
206 //::new(t)MixedDimensionMesh<ELEMENT_DIM, SPACE_DIM>(partition_type);
207 ::new(t)MixedDimensionMesh<ELEMENT_DIM, SPACE_DIM>(DistributedTetrahedralMeshPartitionType::DUMB);
208
209 /*
210 * The exception needs to be thrown after the call to ::new(t), or Boost will try
211 * to free non-allocated memory when the exception is thrown.
212 */
213 if (DistributedVectorFactory::CheckNumberOfProcessesOnLoad() &&
214 num_procs != PetscTools::GetNumProcs())
215 {
216 EXCEPTION("This archive was written for a different number of processors");
217 }

219}

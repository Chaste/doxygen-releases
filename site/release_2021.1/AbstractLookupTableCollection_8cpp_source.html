63 void AbstractLookupTableCollection::SetTableProperties(const std::string& rKeyingVariableName, double min, double step, double max)
64 {
65  // Check inputs
66  unsigned num_steps = (unsigned) ((max-min)/step+0.5);
68  if (fabs(max - min - num_steps*step) > 1e-10)
69  {
70  EXCEPTION("Table step size does not divide range between table limits.");
71  }
72  // Set state
73  unsigned i = GetTableIndex(rKeyingVariableName);
74  if ((min != mTableMins[i]) || (step != mTableSteps[i]) || (max != mTableMaxs[i]))
75  {
76  mNeedsRegeneration[i] = true;
77  }
78  mTableMins[i] = min;
79  mTableSteps[i] = step;
80  mTableStepInverses[i] = 1/step;
81  mTableMaxs[i] = max;
82 }

93 unsigned AbstractLookupTableCollection::GetTableIndex(const std::string& rKeyingVariableName) const
94 {
95  unsigned i=0;
96  for (; i<mKeyingVariableNames.size(); i++)
97  {
98  if (mKeyingVariableNames[i] == rKeyingVariableName)
99  {
100  break;
101  }
102  }
103  if (i == mKeyingVariableNames.size())
104  {
105  EXCEPTION("Lookup table keying variable '" + rKeyingVariableName + "' does not exist.");
106  }
107  return i;
108 }

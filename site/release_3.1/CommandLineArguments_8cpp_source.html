00117 int CommandLineArguments::GetNumberOfArgumentsForOption(std::string option)
00118 {
00119     int start_idx = GetIndexForArgument(option);
00120     if (start_idx < 0)
00121     {
00122         EXCEPTION("Command line option '" + option + "' does not exist");
00123     }
00124 
00125     int end_idx = start_idx;
00126     for (int i=start_idx+1; i<*p_argc; i++)
00127     {
00128         std::string argument = std::string((*p_argv)[i]);
00129         if (argument.substr(0,1)=="-" && argument.substr(1,1).find_first_of("0123456789")==std::string::npos)
00130         {
00131             break;
00132         }
00133         end_idx = i;
00134     }
00135 
00136     if (end_idx == start_idx)
00137     {
00138         EXCEPTION("No value(s) given after command line option '" + option + "'");
00139     }
00140 
00141     return end_idx - start_idx;
00142 }

61 template<unsigned DIM>
62 void PlaneBasedCellKiller<DIM>::CheckAndLabelCellsForApoptosisOrDeath()
63 {
64  for (typename AbstractCellPopulation<DIM>::Iterator cell_iter = this->mpCellPopulation->Begin();
65  cell_iter != this->mpCellPopulation->End();
66  ++cell_iter)
67  {
68  c_vector<double, DIM> cell_location = this->mpCellPopulation->GetLocationOfCellCentre(*cell_iter);
69 
70  if (inner_prod(cell_location - mPointOnPlane, mNormalToPlane) > 0.0)
71  {
72  cell_iter->Kill();
73  }
74  }
75 }

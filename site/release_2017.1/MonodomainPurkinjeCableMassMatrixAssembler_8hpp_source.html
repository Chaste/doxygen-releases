96  for(unsigned i=0; i<2; i++) // 2 = number of basis functions per cable element

102  ret(2*i, 2*j+1) = 0; // [V,Vpurkinje] block
103  ret(2*i+1,2*j+1) = rPhi(i)*rPhi(j);
104 
105  ret(2*i+1,2*j+1) *= fibre_cross_section_area;
106  }

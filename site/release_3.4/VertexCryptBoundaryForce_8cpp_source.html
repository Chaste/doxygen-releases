53 template<unsigned DIM>
54 void VertexCryptBoundaryForce<DIM>::AddForceContribution(AbstractCellPopulation<DIM>& rCellPopulation)
55 {
56  // Helper variable that is a static cast of the cell population
57  VertexBasedCellPopulation<DIM>* p_cell_population = static_cast<VertexBasedCellPopulation<DIM>*>(&rCellPopulation);
58 
59  // Throw an exception message if not using a VertexBasedCellPopulation
60  if (dynamic_cast<VertexBasedCellPopulation<DIM>*>(&rCellPopulation) == NULL)
61  {
62  EXCEPTION("VertexCryptBoundaryForce is to be used with VertexBasedCellPopulations only");
63  }
64 
65  // Iterate over nodes
66  for (typename AbstractMesh<DIM,DIM>::NodeIterator node_iter = p_cell_population->rGetMesh().GetNodeIteratorBegin();
67  node_iter != p_cell_population->rGetMesh().GetNodeIteratorEnd();
68  ++node_iter)
69  {
70  double y = node_iter->rGetLocation()[1]; // y-coordinate of node
71 
72  // If the node lies below the line y=0, then add the boundary force contribution to the node forces
73  if (y < 0.0)
74  {
75  c_vector<double, DIM> boundary_force = zero_vector<double>(DIM);
76  boundary_force[1] = mForceStrength*SmallPow(y, 2);
77 
78  node_iter->AddAppliedForceContribution(boundary_force);
79  }
80  }
81 }

115void AbstractMeshWriter<ELEMENT_DIM, SPACE_DIM>::WriteFilesUsingMeshReader(AbstractMeshReader<ELEMENT_DIM, SPACE_DIM>& rMeshReader)
116{
117 mpMeshReader = &rMeshReader;
118 mNumNodes = mpMeshReader->GetNumNodes();
119 mNumElements = mpMeshReader->GetNumElements();
120 mNumBoundaryElements = mpMeshReader->GetNumFaces();
121
122 // Only triangles mesh readers know about cable elements
123 mNumCableElements = mpMeshReader->GetNumCableElements();
124
125 if (PetscTools::AmMaster())
126 {
127 WriteFiles();
128 }
129}

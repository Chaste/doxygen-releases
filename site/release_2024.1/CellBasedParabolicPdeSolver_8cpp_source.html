51template<unsigned DIM>
52c_vector<double, 1*(DIM+1)> CellBasedParabolicPdeSolver<DIM>::ComputeVectorTerm(
53 c_vector<double, DIM+1>& rPhi,
54 c_matrix<double, DIM, DIM+1>& rGradPhi,
55 ChastePoint<DIM>& rX,
56 c_vector<double, 1>& rU,
57 c_matrix<double, 1, DIM>& rGradU /* not used */,
58 Element<DIM, DIM>* pElement)
59{
60 return (mInterpolatedSourceTerm
61 + PdeSimulationTime::GetPdeTimeStepInverse() * this->mpParabolicPde->ComputeDuDtCoefficientFunction(rX) * rU(0)) * rPhi;
62}

64template<unsigned DIM>
65c_matrix<double, 1*(DIM+1), 1*(DIM+1)> CellBasedParabolicPdeSolver<DIM>::ComputeMatrixTerm(
66 c_vector<double, DIM+1>& rPhi,
67 c_matrix<double, DIM, DIM+1>& rGradPhi,
68 ChastePoint<DIM>& rX,
69 c_vector<double, 1>& rU,
70 c_matrix<double, 1, DIM>& rGradU,
71 Element<DIM, DIM>* pElement)
72{
73
74 c_matrix<double, DIM, DIM> pde_diffusion_term = this->mpParabolicPde->ComputeDiffusionTerm(rX, pElement);
75
76 return prod( trans(rGradPhi), c_matrix<double, DIM, DIM+1>(prod(pde_diffusion_term, rGradPhi)) )
77 + PdeSimulationTime::GetPdeTimeStepInverse() * this->mpParabolicPde->ComputeDuDtCoefficientFunction(rX) * outer_prod(rPhi, rPhi);
78}

86template<unsigned DIM>
87void CellBasedParabolicPdeSolver<DIM>::IncrementInterpolatedQuantities(double phiI, const Node<DIM>* pNode)
88{
89 unsigned index_of_unknown = 0;
90 double u_at_node = this->GetCurrentSolutionOrGuessValue(pNode->GetIndex(), index_of_unknown);
91
92 mInterpolatedSourceTerm += phiI*this->mpParabolicPde->ComputeSourceTermAtNode(*pNode,u_at_node);
93}

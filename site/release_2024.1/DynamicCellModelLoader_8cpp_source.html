51DynamicCellModelLoader::DynamicCellModelLoader(const std::string& rLoadableModulePath)
52 : mLoadableModulePath(rLoadableModulePath)
53{
54 // Open the .so file
55 mpDynamicModule = dlopen(rLoadableModulePath.c_str(), RTLD_NOW);
56 if (mpDynamicModule == NULL)
57 {
58 EXCEPTION("Unable to load .so file '" + rLoadableModulePath + "': " + dlerror());
59 }
60
61 // Find the cell creation function
62 dlerror(); // Reset errors
63 void* p_creation_function = dlsym(mpDynamicModule, "MakeCardiacCell");
64 // Check it exists
65 const char* p_error = dlerror();
66 if (p_error)
67 {
68 EXCEPTION("Failed to load cell creation function from .so file '" + rLoadableModulePath
69 + "': " + p_error);
70 }
71 // Cast to the right type
72 mpCreationFunction = reinterpret_cast<CellCreationFunctionType*>(p_creation_function);
73}

75DynamicCellModelLoader::~DynamicCellModelLoader()
76{
77 if (mpDynamicModule)
78 {
79 dlclose(mpDynamicModule);
80 }
81}

83AbstractCardiacCellInterface* DynamicCellModelLoader::CreateCell(boost::shared_ptr<AbstractIvpOdeSolver> pSolver,
84 boost::shared_ptr<AbstractStimulusFunction> pStimulus)
85{
86 AbstractCardiacCellInterface* p_cell = (*mpCreationFunction)(pSolver, pStimulus);
87
88 AbstractDynamicallyLoadableEntity* p_entity = dynamic_cast<AbstractDynamicallyLoadableEntity*>(p_cell);
89 assert(p_entity != NULL);
90 p_entity->SetLoader(shared_from_this());
91
92 return p_cell;
93}

40 SimpleDataWriter::SimpleDataWriter(const std::string& rDirectory,
41  const std::string& rFileName,
42  const std::vector<std::vector<double> >& rData,
43  bool cleanDirectory)
44 {
45  if (rData.size() == 0)
46  {
47  EXCEPTION("Data vector is empty");
48  }
49 
50  for (unsigned i=0; i<rData.size(); i++)
51  {
52  if (rData[i].size() != rData[0].size())
53  {
54  EXCEPTION("Data vector sizes are not all equal");
55  }
56  }
57 
58  OutputFileHandler output_file_handler(rDirectory, cleanDirectory);
59  out_stream p_file = output_file_handler.OpenOutputFile(rFileName);
60 
61  for (unsigned j=0; j<rData[0].size(); j++)
62  {
63  for (unsigned i=0; i<rData.size(); i++)
64  {
65  (*p_file) << rData[i][j] << "\t";
66  }
67  (*p_file) << "\n";
68  }
69  p_file->close();
70 }

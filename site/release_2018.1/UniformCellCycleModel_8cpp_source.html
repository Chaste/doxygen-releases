70 void UniformCellCycleModel::SetCellCycleDuration()
71 {
72  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
73 
74  if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
75  {
76  mCellCycleDuration = DBL_MAX;
77  }
78  else
79  {
80  mCellCycleDuration = mMinCellCycleDuration + (mMaxCellCycleDuration - mMinCellCycleDuration) * p_gen->ranf(); // U[MinCCD,MaxCCD]
81  }
82 }

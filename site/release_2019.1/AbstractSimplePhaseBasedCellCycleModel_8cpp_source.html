77 void AbstractSimplePhaseBasedCellCycleModel::SetG1Duration()
78 {
79  assert(mpCell != nullptr);
80 
81  if (mpCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>())
82  {
83  mG1Duration = GetStemCellG1Duration();
84  }
85  else if (mpCell->GetCellProliferativeType()->IsType<TransitCellProliferativeType>())
86  {
87  mG1Duration = GetTransitCellG1Duration();
88  }
89  else if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
90  {
91  mG1Duration = DBL_MAX;
92  }
93  else
94  {
95  NEVER_REACHED;
96  }
97 }

105 void AbstractSimplePhaseBasedCellCycleModel::UpdateCellCyclePhase()
106 {
107  double time_since_birth = GetAge();
108  assert(time_since_birth >= 0);
109 
110  if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
111  {
112  mCurrentCellCyclePhase = G_ZERO_PHASE;
113  }
114  else if (time_since_birth < GetMDuration())
115  {
116  mCurrentCellCyclePhase = M_PHASE;
117  }
118  else if (time_since_birth < GetMDuration() + mG1Duration)
119  {
120  mCurrentCellCyclePhase = G_ONE_PHASE;
121  }
122  else if (time_since_birth < GetMDuration() + mG1Duration + GetSDuration())
123  {
124  mCurrentCellCyclePhase = S_PHASE;
125  }
126  else if (time_since_birth < GetMDuration() + mG1Duration + GetSDuration() + GetG2Duration())
127  {
128  mCurrentCellCyclePhase = G_TWO_PHASE;
129  }
130 }

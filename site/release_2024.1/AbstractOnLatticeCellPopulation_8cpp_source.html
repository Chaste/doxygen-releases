38template<unsigned DIM>
39AbstractOnLatticeCellPopulation<DIM>::AbstractOnLatticeCellPopulation(AbstractMesh<DIM, DIM>& rMesh,
40 std::vector<CellPtr>& rCells,
41 const std::vector<unsigned> locationIndices,
42 bool deleteMesh)
43 : AbstractCellPopulation<DIM>(rMesh, rCells, locationIndices),
44 mDeleteMesh(deleteMesh),
45 mUpdateNodesInRandomOrder(true),
46 mIterateRandomlyOverUpdateRuleCollection(false)
47{
48 std::list<CellPtr>::iterator it = this->mCells.begin();
49 for (unsigned i=0; it != this->mCells.end(); ++it, ++i)
50 {
51 unsigned index = locationIndices.empty() ? i : locationIndices[i]; // assume that the ordering matches
52 AbstractCellPopulation<DIM>::AddCellUsingLocationIndex(index, *it);
53 }
54}

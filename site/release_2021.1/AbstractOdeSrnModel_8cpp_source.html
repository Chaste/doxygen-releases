71 void AbstractOdeSrnModel::SimulateToCurrentTime()
72 {
73  assert(mpOdeSystem != nullptr);
74  assert(SimulationTime::Instance()->IsStartTimeSetUp());
75 
76  double current_time = SimulationTime::Instance()->GetTime();
77 
78  // Run ODEs if needed
79  if (current_time > mLastTime)
80  {
81  if (!this->mFinishedRunningOdes)
82  {
83  // Update whether a stopping event has occurred
84  this->mFinishedRunningOdes = SolveOdeToTime(current_time);
85  }
86  else
87  {
88  // ODE model finished, just increasing time...
89  }
90  }
91 
92  // Update the SimulatedToTime value
93  mLastTime = current_time;
94  SetSimulatedToTime(current_time);
95 }

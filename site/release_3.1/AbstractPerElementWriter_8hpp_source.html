00104     void WriteData(OutputFileHandler& rHandler, const std::string& rFileName, AbstractTetrahedralMesh<ELEMENT_DIM, SPACE_DIM>* pMesh)
00105     {
00106         mpMesh = pMesh;
00107         c_vector<double, DATA_SIZE> data;
00108         if (PetscTools::AmMaster())
00109         {
00110             mpMasterFile = rHandler.OpenOutputFile(rFileName);
00111             MPI_Status status;
00112             status.MPI_ERROR = MPI_SUCCESS; //For MPICH2
00113             WriteHeaderOnMaster();
00114             for (unsigned element_index=0; element_index<mpMesh->GetNumElements();element_index++)
00115             {
00116                 if (mpMesh->CalculateDesignatedOwnershipOfElement(element_index))
00117                 {
00118                     //Master owns this process and can write it directly
00119                     Visit(mpMesh->GetElement(element_index), data);
00120                 }
00121                 else
00122                 {
00123                     //Data must come from a remote process
00124                     MPI_Recv(&data[0], DATA_SIZE, MPI_DOUBLE, MPI_ANY_SOURCE, element_index, PETSC_COMM_WORLD, &status);
00125                 }
00126                 WriteElementOnMaster(data);
00127             }
00128             *mpMasterFile << "# "<<ChasteBuildInfo::GetProvenanceString();
00129             mpMasterFile->close();
00130         }
00131         else
00132         {
00133             //Not master process
00134             unsigned previous_index = 0u; //Used to check that the indices are monotone
00135             for (typename AbstractTetrahedralMesh<ELEMENT_DIM,SPACE_DIM>::ElementIterator iter = mpMesh->GetElementIteratorBegin();
00136                           iter != mpMesh->GetElementIteratorEnd();
00137                           ++iter)
00138             {
00139                 unsigned element_index = iter->GetIndex();
00140                 //Check monotonicity
00141                 if (previous_index>0u)
00142                 {
00143                     assert(element_index > previous_index);
00144                 }
00145                 previous_index = element_index;
00146                 if (mpMesh->CalculateDesignatedOwnershipOfElement(element_index))
00147                 {
00148                     //The master needs to know about this one
00149                     Visit(&(*iter), data);
00150                     MPI_Send(&data[0], DATA_SIZE, MPI_DOUBLE, 0, element_index, PETSC_COMM_WORLD);//Tag with element_index
00151                 }
00152             }
00153 
00154         }
00155     }

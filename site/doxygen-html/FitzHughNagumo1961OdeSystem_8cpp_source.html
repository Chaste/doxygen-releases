61void FitzHughNagumo1961OdeSystem::EvaluateYDerivatives(double time, const std::vector<double> &rY, std::vector<double>& rDY)
62{
63 double membrane_V = rY[0]; // v
64 double recovery_variable = rY[1]; // w
65
66 double i_stim = GetIntracellularAreaStimulus(time);
67
68 // dV/dt
69 double membrane_V_prime = 0;
70 // do not update voltage if the mSetVoltageDerivativeToZero flag has been set
71 if (!mSetVoltageDerivativeToZero)
72 {
73 membrane_V_prime = membrane_V*(membrane_V-mAlpha)*(1-membrane_V)-recovery_variable+i_stim;
74 }
75
76 // dw/dt
77 double recovery_variable_prime = mEpsilon*(membrane_V-mGamma*recovery_variable);
78
79 rDY[0] = membrane_V_prime;
80 rDY[1] = recovery_variable_prime;
81}

83double FitzHughNagumo1961OdeSystem::GetIIonic(const std::vector<double>* pStateVariables)
84{
85 if (!pStateVariables) pStateVariables = &mStateVariables;
86 double membrane_V = (*pStateVariables)[mVoltageIndex];
87 double recovery_variable = (*pStateVariables)[1];
88 double fake_ionic_current = membrane_V*(membrane_V-mAlpha)*(1-membrane_V)-recovery_variable;
89 return fake_ionic_current;
90}

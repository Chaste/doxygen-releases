40void
41SingleTraceOutputModifier::InitialiseAtStart(DistributedVectorFactory* pVectorFactory, const std::vector<unsigned>& rNodePermutation)
42{
43 // Collectively open the output directory - this might already be in place from creating the HDF5 file
44 OutputFileHandler output_handler(HeartConfig::Instance()->GetOutputDirectory(), false);
45
46 if (!rNodePermutation.empty()){
47 //Convert to runtime permuted index
48 mGlobalIndex = rNodePermutation[mGlobalIndex];
49 }
50 // Note that the runtime index is the one which will get archived
51
52 // Only the node owner needs to proceed here
53 if (pVectorFactory->IsGlobalIndexLocal(mGlobalIndex))
54 {
55 // Set the local index correctly
56 mLocalIndex = mGlobalIndex - pVectorFactory->GetLow();
57 // Open the file as new
58 mFileStream = output_handler.OpenOutputFile(mFilename);
59 }
60}

72void
73SingleTraceOutputModifier::ProcessSolutionAtTimeStep(double time, Vec solution, unsigned problemDim)
74{
75 if (mLocalIndex != UINT_MAX)
76 {
77 double* p_solution;
78 VecGetArray(solution, &p_solution); //This does not need to be collective
79 (*mFileStream) << time;
80 for (unsigned i=0; i<problemDim; ++i)
81 {
82 (*mFileStream) << "\t" << p_solution[mLocalIndex*problemDim+i];
83 }
84 (*mFileStream) << "\n";
85 VecRestoreArray(solution, &p_solution);
86
87 if (mFlushTime > 0.0 && Divides(mFlushTime, time))
88 {
89 mFileStream->flush();
90 }
91 }
92}

49template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
50void ForwardEulerNumericalMethod<ELEMENT_DIM,SPACE_DIM>::UpdateAllNodePositions(double dt)
51{
52 if (!this->mUseUpdateNodeLocation)
53 {
54 // Apply forces to each cell, and save a vector of net forces F
55 std::vector<c_vector<double, SPACE_DIM> > forces = this->ComputeForcesIncludingDamping();
56
57 unsigned index = 0;
58 for (typename AbstractMesh<ELEMENT_DIM, SPACE_DIM>::NodeIterator node_iter = this->mpCellPopulation->rGetMesh().GetNodeIteratorBegin();
59 node_iter != this->mpCellPopulation->rGetMesh().GetNodeIteratorEnd();
60 ++node_iter, ++index)
61 {
62 // Get the current node location and calculate the new location according to the forward Euler method
63 const c_vector<double, SPACE_DIM>& r_old_location = node_iter->rGetLocation();
64 c_vector<double, SPACE_DIM> displacement = dt * forces[index];
65
66 // In the vertex-based case, the displacement may be scaled if the cell rearrangement threshold is exceeded
67 this->DetectStepSizeExceptions(node_iter->GetIndex(), displacement, dt);
68
69 c_vector<double, SPACE_DIM> new_location = r_old_location + displacement;
70 this->SafeNodePositionUpdate(node_iter->GetIndex(), new_location);
71 }
72 }
73 else
74 {
75 /*
76 * If this type of cell population does not support the new numerical methods, delegate
77 * updating node positions to the population itself.
78 *
79 * This only applies to NodeBasedCellPopulationWithBuskeUpdates.
80 */
81 this->mpCellPopulation->UpdateNodeLocations(dt);
82 }
83}

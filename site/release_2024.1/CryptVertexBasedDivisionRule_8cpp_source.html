40template <unsigned SPACE_DIM>
41c_vector<double, SPACE_DIM> CryptVertexBasedDivisionRule<SPACE_DIM>::CalculateCellDivisionVector(
42 CellPtr pParentCell,
43 VertexBasedCellPopulation<SPACE_DIM>& rCellPopulation)
44{
45 assert(SPACE_DIM == 2); // LCOV_EXCL_LINE
46
47 c_vector<double, SPACE_DIM> axis_of_division;
48
49 double random_angle = 2.0*M_PI*RandomNumberGenerator::Instance()->ranf();
50 axis_of_division(0) = cos(random_angle);
51 axis_of_division(1) = sin(random_angle);
52
53 // We don't need to prescribe how 'stem' cells divide if Wnt is not present
54 bool is_wnt_included = WntConcentration<2>::Instance()->IsWntSetUp();
55 if (!is_wnt_included)
56 {
57 WntConcentration<2>::Destroy();
58 if (pParentCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>())
59 {
60 axis_of_division(0) = 1.0;
61 axis_of_division(1) = 0.0;
62 }
63 }
64
65 return axis_of_division;
66}

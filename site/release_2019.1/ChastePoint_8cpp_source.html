114 template<unsigned DIM>
115 bool ChastePoint<DIM>::IsSamePoint(const ChastePoint<DIM>& rPoint) const
116 {
117  bool returned_value = true;
118  for (unsigned dim=0; dim<DIM; dim++)
119  {
120  if (rPoint[dim] != mLocation[dim])

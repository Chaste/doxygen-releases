39 S1S2Stimulus::S1S2Stimulus(double magnitude, double stimulusDuration, double s1Duration, double s1Period, double startTime, std::vector<double> s2Periods)
40  : MultiStimulus()
41 {
42  mNumS2FrequencyValues = s2Periods.size();
43  for (unsigned i=0; i<mNumS2FrequencyValues; i++)
44  {
45  boost::shared_ptr<MultiStimulus> p_experiment_stimulus(new MultiStimulus());
46 
47  // The 0.5*(s1Period-stimulusDuration) taken off the stop time is to avoid getting a double stimulus at the instant s1Duration+startTime (#2442).
48  boost::shared_ptr<RegularStimulus> p_s1(new RegularStimulus(magnitude, stimulusDuration, s1Period, startTime, s1Duration+startTime-0.5*(s1Period-stimulusDuration)));
49  boost::shared_ptr<RegularStimulus> p_s2(new RegularStimulus(magnitude, stimulusDuration, s2Periods[i], s1Duration+startTime, s1Duration + 2*s2Periods[i]+startTime));
50 
51  p_experiment_stimulus->AddStimulus(p_s1);
52  p_experiment_stimulus->AddStimulus(p_s2);
53 
54  // We hijack the mStimulus of the MultiStimulus class to
55  // record different stimuli for each S2 frequency.
56  this->AddStimulus(p_experiment_stimulus);
57  }
58  mS2Index=0u;
59 }

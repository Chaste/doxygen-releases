49 AbstractCellCycleModel* GammaDistributedStochasticDurationCellCycleModel::CreateCellCycleModel()
50 {
51  // Create a new cell-cycle model
52  GammaDistributedStochasticDurationCellCycleModel* p_model = new GammaDistributedStochasticDurationCellCycleModel();
53 
54  /*
55  * Set each member variable of the new cell-cycle model that inherits
56  * its value from the parent.
57  *
58  * Note 1: some of the new cell-cycle model's member variables (namely
59  * mBirthTime, mCurrentCellCyclePhase, mReadyToDivide) will already have been
60  * correctly initialized in its constructor.
61  *
62  * Note 2: one or more of the new cell-cycle model's member variables
63  * may be set/overwritten as soon as InitialiseDaughterCell() is called on
64  * the new cell-cycle model.
65  *
66  * Note 3: the member variable mDimension remains unset, since this cell-cycle
67  * model does not need to know the spatial dimension, so if we were to call
68  * SetDimension() on the new cell-cycle model an exception would be triggered;
69  * hence we do not set this member variable.
70  */
71  p_model->SetBirthTime(mBirthTime);
72  p_model->SetMinimumGapDuration(mMinimumGapDuration);
73  p_model->SetStemCellG1Duration(mStemCellG1Duration);
74  p_model->SetTransitCellG1Duration(mTransitCellG1Duration);
75  p_model->SetSDuration(mSDuration);
76  p_model->SetG2Duration(mG2Duration);
77  p_model->SetMDuration(mMDuration);
78  p_model->SetShape(mShape);
79  p_model->SetScale(mScale);
80 
81  return p_model;
82 }

84 void GammaDistributedStochasticDurationCellCycleModel::SetG1Duration()
85 {
86  if ( mpCell->GetCellProliferativeType()->IsType<StemCellProliferativeType>()
87  || mpCell->GetCellProliferativeType()->IsType<TransitCellProliferativeType>() )
88  {
89  // Generate a gamma random number with mShape and mScale
90  mG1Duration = RandomNumberGenerator::Instance()->GammaRandomDeviate(mShape, mScale);
91  }
92  else if (mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>())
93  {
94  mG1Duration = DBL_MAX;
95  }
96  else
97  {
98  NEVER_REACHED;
99  }
100 }

1098template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
1099AbstractCellPopulation<ELEMENT_DIM, SPACE_DIM>::Iterator::Iterator(AbstractCellPopulation& rCellPopulation, std::list<CellPtr>::iterator cellIter)
1100 : mrCellPopulation(rCellPopulation),

1115 }

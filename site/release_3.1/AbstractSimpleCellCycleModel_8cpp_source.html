00085 void AbstractSimpleCellCycleModel::UpdateCellCyclePhase()
00086 {
00087     double time_since_birth = GetAge();
00088     assert(time_since_birth >= 0);
00089 
00090     if (mpCell->GetCellProliferativeType() == DIFFERENTIATED)
00091     {
00092         mCurrentCellCyclePhase = G_ZERO_PHASE;
00093     }
00094     else if (time_since_birth < GetMDuration())
00095     {
00096         mCurrentCellCyclePhase = M_PHASE;
00097     }
00098     else if (time_since_birth < GetMDuration() + mG1Duration)
00099     {
00100         mCurrentCellCyclePhase = G_ONE_PHASE;
00101     }
00102     else if (time_since_birth < GetMDuration() + mG1Duration + GetSDuration())
00103     {
00104         mCurrentCellCyclePhase = S_PHASE;
00105     }
00106     else if (time_since_birth < GetMDuration() + mG1Duration + GetSDuration() + GetG2Duration())
00107     {
00108         mCurrentCellCyclePhase = G_TWO_PHASE;
00109     }
00110 }

71 template<unsigned DIM>
72 double CellwiseSourceParabolicPde<DIM>::ComputeSourceTermAtNode(const Node<DIM>& rNode, double u)
73 {
74  double source_coefficient = 0.0;
75 
76  if (mrCellPopulation.IsPdeNodeAssociatedWithNonApoptoticCell(rNode.GetIndex()))
77  {
78  source_coefficient = mSourceCoefficient;
79  }
80 
81  // The source term is C*u

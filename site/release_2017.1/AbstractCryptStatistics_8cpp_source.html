50 void AbstractCryptStatistics::LabelSPhaseCells()
51 {
52  for (AbstractCellPopulation<2>::Iterator cell_iter = mrCrypt.Begin();
53  cell_iter != mrCrypt.End();
54  ++cell_iter)
55  {
56  if (static_cast <AbstractPhaseBasedCellCycleModel*>(cell_iter->GetCellCycleModel())->GetCurrentCellCyclePhase()== S_PHASE)
57  {
58  // This should only be done for wild type cells (at the moment anyway)
59  assert(cell_iter->GetMutationState()->IsType<WildTypeCellMutationState>());
60 
61  // Label this cell
62  if (!cell_iter->HasCellProperty<CellLabel>())
63  {
64  /*
65  * This method is usually called after mrCrypt has called CellPropertyRegistry::TakeOwnership().
66  * This means that were we to call CellPropertyRegistry::Instance() here when adding the CellLabel,
67  * we would be creating a new CellPropertyRegistry. In this case the CellLabel's cell count would
68  * be incorrect. We therefore access the CellLabel via mrCrypt.GetCellPropertyRegistry().
69  */
70  cell_iter->AddCellProperty(mrCrypt.GetCellPropertyRegistry()->Get<CellLabel>());
71  }
72  }
73  }
74 }

57template<class CARDIAC_CELL>
58boost::shared_ptr<AbstractModifier> AbstractCardiacCellWithModifiers<CARDIAC_CELL>::GetModifier(const std::string& rModifierName)
59{
60 if (mModifiersMap.find(rModifierName) == mModifiersMap.end())
61 {
62 EXCEPTION("There is no modifier called " + rModifierName + " in this model.");
63 }
64 return *(mModifiersMap[rModifierName]);
65}

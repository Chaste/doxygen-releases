49template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
50c_vector<double, SPACE_DIM> CellAppliedForceWriter<ELEMENT_DIM, SPACE_DIM>::GetVectorCellDataForVtkOutput(CellPtr pCell, AbstractCellPopulation<ELEMENT_DIM, SPACE_DIM>* pCellPopulation)
51{
52 c_vector<double, SPACE_DIM> applied_force = scalar_vector<double>(SPACE_DIM, DOUBLE_UNSET);
53
54 if (dynamic_cast<NodeBasedCellPopulation<SPACE_DIM>*>(pCellPopulation))
55 {
56 unsigned node_index = pCellPopulation->GetLocationIndexUsingCell(pCell);
57 applied_force = pCellPopulation->GetNode(node_index)->rGetAppliedForce();
58 }
59
60 return applied_force;
61}

63template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
64void CellAppliedForceWriter<ELEMENT_DIM, SPACE_DIM>::VisitCell(CellPtr pCell, AbstractCellPopulation<ELEMENT_DIM, SPACE_DIM>* pCellPopulation)
65{
66 unsigned location_index = pCellPopulation->GetLocationIndexUsingCell(pCell);
67 unsigned cell_id = pCell->GetCellId();
68 c_vector<double, SPACE_DIM> cell_location = pCellPopulation->GetLocationOfCellCentre(pCell);
69 c_vector<double, SPACE_DIM> applied_force = GetVectorCellDataForVtkOutput(pCell, pCellPopulation);
70
71 *this->mpOutStream << location_index << " " << cell_id << " ";
72 for (unsigned i=0; i<SPACE_DIM; i++)
73 {
74 *this->mpOutStream << cell_location[i] << " ";
75 }
76 for (unsigned i=0; i<SPACE_DIM; i++)
77 {
78 *this->mpOutStream << applied_force[i] << " ";
79 }
80}

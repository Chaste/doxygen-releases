00039 template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
00040 AbstractHdf5Converter<ELEMENT_DIM, SPACE_DIM>::AbstractHdf5Converter(std::string inputDirectory,
00041                                                                      std::string fileBaseName,
00042                                                                      AbstractTetrahedralMesh<ELEMENT_DIM, SPACE_DIM>* pMesh,
00043                                                                      std::string subdirectoryName)
00044     : mFileBaseName(fileBaseName),
00045       mpMesh(pMesh),
00046       mRelativeSubdirectory(subdirectoryName)
00047 {
00048     // Store directory, mesh and filenames and create the reader
00049     mpReader = new Hdf5DataReader(inputDirectory, mFileBaseName);
00050 
00051     // Create new directory in which to store everything
00052     mpOutputFileHandler = new OutputFileHandler(inputDirectory + "/" + mRelativeSubdirectory, false);
00053 
00054     // Check the data file for basic validity
00055     std::vector<std::string> variable_names = mpReader->GetVariableNames();
00056     mNumVariables = variable_names.size();
00057 
00058     if (mpReader->GetNumberOfRows() != mpMesh->GetNumNodes())
00059     {
00060         delete mpReader;
00061         delete mpOutputFileHandler;
00062         EXCEPTION("Mesh and HDF5 file have a different number of nodes");
00063     }
00064 
00065     // Write an info file
00066     if (PetscTools::AmMaster())
00067     {
00068         // Note that we don't want the child processes to write info files
00069         out_stream p_file = mpOutputFileHandler->OpenOutputFile(mFileBaseName + "_times.info");
00070 
00071         std::vector<double> time_values = mpReader->GetUnlimitedDimensionValues();
00072         unsigned num_timesteps = time_values.size();
00073         double first_timestep = time_values.front();
00074         double last_timestep = time_values.back();
00075 
00076         double timestep = num_timesteps > 1 ? time_values[1] - time_values[0] : DOUBLE_UNSET;
00077 
00078         *p_file << "Number of timesteps " << num_timesteps << std::endl;
00079         *p_file << "timestep " << timestep << std::endl;
00080         *p_file << "First timestep " << first_timestep << std::endl;
00081         *p_file << "Last timestep " << last_timestep << std::endl;
00082         *p_file << ChasteBuildInfo::GetProvenanceString();
00083 
00084         p_file->close();
00085     }
00086 }

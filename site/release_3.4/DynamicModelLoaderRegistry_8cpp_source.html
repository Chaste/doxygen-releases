48 DynamicCellModelLoaderPtr DynamicModelLoaderRegistry::GetLoader(const std::string& rPath)
49 {
50  // Delete any waiting loaders
52  //mDeletableLoaders.clear();
53 
54  DynamicCellModelLoaderPtr p_loader;
55  // Have we opened this library already?
56  std::map<std::string, DynamicCellModelLoaderWeakPtr>::iterator it = mLoaders.find(rPath);
57  if (it == mLoaders.end() || it->second.expired())
58  {
59  // Not loaded, or no longer used
60  p_loader = DynamicCellModelLoader::Create(rPath);
61  mLoaders[rPath] = p_loader;
62  }
63  else
64  {
65  p_loader = it->second.lock();
66  }
67  ScheduleForDeletion(p_loader);
68  return p_loader;

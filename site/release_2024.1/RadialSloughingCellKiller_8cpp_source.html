55void RadialSloughingCellKiller::CheckAndLabelCellsForApoptosisOrDeath()
56{
57 for (AbstractCellPopulation<2>::Iterator cell_iter = this->mpCellPopulation->Begin();
58 cell_iter != this->mpCellPopulation->End();
59 ++cell_iter)
60 {
61 // Get distance from centre of cell population
62 double r = norm_2(this->mpCellPopulation->GetLocationOfCellCentre(*cell_iter) - mCentre);
63
64 if (r > mRadius)
65 {
66 // Mark the cell as killed and store removal information if required.
67 this->mpCellPopulation->KillCell(*cell_iter, "RadialSloughingCellKiller");
68 }
69 }
70}

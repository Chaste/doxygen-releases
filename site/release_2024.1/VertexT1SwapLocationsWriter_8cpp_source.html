71template<unsigned ELEMENT_DIM, unsigned SPACE_DIM>
72void VertexT1SwapLocationsWriter<ELEMENT_DIM, SPACE_DIM>::Visit(VertexBasedCellPopulation<SPACE_DIM>* pCellPopulation)
73{
74 std::vector<T1SwapInfo<SPACE_DIM> > t1_swap_info
75 = pCellPopulation->rGetMesh().GetOperationRecorder()->GetT1SwapsInfo();
76 *this->mpOutStream << t1_swap_info.size() << "\t";
77 for (unsigned index = 0; index < t1_swap_info.size(); index++)
78 {
79 for (unsigned i=0; i<SPACE_DIM; i++)
80 {
81 *this->mpOutStream << t1_swap_info[index].mLocation[i] << "\t";
82 }
83 }
84
85 pCellPopulation->rGetMesh().GetOperationRecorder()->ClearT1SwapsInfo();
86}

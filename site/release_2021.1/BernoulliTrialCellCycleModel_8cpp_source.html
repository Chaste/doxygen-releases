60 bool BernoulliTrialCellCycleModel::ReadyToDivide()
61 {
62  assert(mpCell != nullptr);
63 
64  if (!mReadyToDivide)
65  {
66  if (GetAge() > mMinimumDivisionAge)
67  {
68  double dt = SimulationTime::Instance()->GetTimeStep();
69  if (!(mpCell->GetCellProliferativeType()->IsType<DifferentiatedCellProliferativeType>()))
70  {
71  RandomNumberGenerator* p_gen = RandomNumberGenerator::Instance();
72  if (p_gen->ranf() < mDivisionProbability*dt)
73  {
74  mReadyToDivide = true;
75  }
76  }
77  }
78  }
79  return mReadyToDivide;
80 }
